168 std::string generateUUID() {
169 // Create an instance of Xoroshiro128Plus with a seed
170 Xoroshiro128Plus rng; // You can use any seed you prefer
171
172 std::stringstream ss;
173 ss << std::hex;
174
175 // Generate 8 random hexadecimal characters (0-15)
176 for (int i = 0; i < 8; i++) {
177 // Generate a random number between 0 and 15 by taking the result of rng() % 16
178 int number = rng() % 16;
179 ss << number;
180 }

182 std::string uuid = ss.str();
