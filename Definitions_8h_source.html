202inline ModelData extractModelDataSparse(GRBModel *model) {
203 ModelData data;
204 try {
205 // Variables
206 int numVars = model->get(GRB_IntAttr_NumVars);
207 GRBVar *vars = model->getVars();
208
209 // Reserve memory based on the actual number of variables
210 data.ub.reserve(numVars);
211 data.lb.reserve(numVars);
212 data.c.reserve(numVars);
213 data.vtype.reserve(numVars);
214 data.name.reserve(numVars);
215
216 std::vector<double> ub(numVars), lb(numVars), obj(numVars);
217 std::vector<char> vtype(numVars);
218
219 auto ubs = model->get(GRB_DoubleAttr_UB, vars, numVars); // Fetch upper bounds
220 auto lbs = model->get(GRB_DoubleAttr_LB, vars, numVars); // Fetch lower bounds
221 auto objs = model->get(GRB_DoubleAttr_Obj, vars, numVars); // Fetch objective coefficients
222 auto vtypes = model->get(GRB_CharAttr_VType, vars, numVars); // Fetch variable types
223
224 // Copy the values to the vectors
225 data.ub.assign(ubs, ubs + numVars);
226 data.lb.assign(lbs, lbs + numVars);
227 data.c.assign(objs, objs + numVars);
228 data.vtype.assign(vtypes, vtypes + numVars);
229
230 // Constraints
231 int numConstrs = model->get(GRB_IntAttr_NumConstrs);
232 SparseMatrix A_sparse;
233
234 // Reserve memory for constraint matrices, estimating 10 non-zeros per row
235 A_sparse.values.reserve(numConstrs * 10); // Estimate non-zero elements
236 data.b.reserve(numConstrs);
237 data.cname.reserve(numConstrs);
238 data.sense.reserve(numConstrs);
239
240 for (int i = 0; i < numConstrs; ++i) {
241 GRBConstr constr = model->getConstr(i);
242
243 // Get the row corresponding to the constraint
244 GRBLinExpr expr = model->getRow(constr);
245 int exprSize = expr.size();
246
247 // Reserve space for constraint elements to minimize reallocation
248 A_sparse.values.reserve(A_sparse.values.size() + exprSize);
249
250 for (int j = 0; j < exprSize; ++j) {
251 GRBVar var = expr.getVar(j);
252 double coeff = expr.getCoeff(j);
253 int varIndex = var.index();
254
255 // Populate SparseElement for A_sparse
256 A_sparse.insert(i, varIndex, coeff);
257 }
258
259 // Batch fetch Constraint Name, RHS, and Sense (improve by reducing single fetches)
260 data.cname.push_back(constr.get(GRB_StringAttr_ConstrName)); // Fetch constraint names
261 data.b.push_back(constr.get(GRB_DoubleAttr_RHS)); // Fetch RHS values
262 data.sense.push_back(constr.get(GRB_CharAttr_Sense)); // Fetch sense values
263 }
264
265 // Set matrix dimensions and build row_start
266 A_sparse.num_cols = numVars;
267 A_sparse.num_rows = numConstrs;
268 A_sparse.buildRowStart();
269
270 // Store the sparse matrix in data
271 data.A_sparse = std::move(A_sparse);
272
273 } catch (GRBException &e) {
274 std::cerr << "Error code = " << e.getErrorCode() << std::endl;
275 std::cerr << e.getMessage() << std::endl;
276 }
277
278 return data;
279}
