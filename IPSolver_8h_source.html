90 CholmodSolver() {

92 c.useGPU = 0;
93 c.nmethods = 1;
94 c.method[0].ordering = CHOLMOD_METIS;
95 c.postorder = true;
96
97#ifdef AUGMENTED
98 c.supernodal = CHOLMOD_SIMPLICIAL;
99#else
100 c.supernodal = CHOLMOD_SUPERNODAL;
101#endif
102
103 L = nullptr;
104 }
105
106 ~CholmodSolver() {
107 if (L) { cholmod_free_factor(&L, &c); }
108 cholmod_finish(&c);
109 }
110
111 void factorizeMatrix(const Eigen::SparseMatrix<double, Eigen::RowMajor, int> &matrix) {
112 cholmod_sparse *A = viewAsCholmod(matrix);
113
114 if (firstFactorization) {
115 if (L) { cholmod_free_factor(&L, &c); }
116 L = cholmod_analyze(A, &c);
117 firstFactorization = false;
118 }
119
120 cholmod_factorize(A, L, &c);
121 }
122
123 Eigen::VectorXd solve(const Eigen::VectorXd &rhs) {
124 cholmod_dense *b = viewAsCholmod(rhs);
125 cholmod_solve2(CHOLMOD_A, L, b, NULL, &X, NULL, &Y, &E, &c);
126 Eigen::VectorXd result = viewAsEigen(X);
127 return result;
128 }
129
130private:
131 static cholmod_sparse *viewAsCholmod(const Eigen::SparseMatrix<double, Eigen::RowMajor, int> &matrix) {
132 cholmod_sparse *result = new cholmod_sparse;
133 result->nrow = matrix.rows();
134 result->ncol = matrix.cols();
135 result->p = const_cast<int *>(matrix.outerIndexPtr());
136 result->i = const_cast<int *>(matrix.innerIndexPtr());
137 result->x = const_cast<double *>(matrix.valuePtr());
138 result->z = nullptr;
139 result->stype = -1;
140 result->itype = CHOLMOD_INT;
141 result->xtype = CHOLMOD_REAL;
142 result->dtype = CHOLMOD_DOUBLE;
143 result->sorted = 1;
144 result->packed = 1;
145 return result;
146 }
147
148 static cholmod_dense *viewAsCholmod(const Eigen::VectorXd &vector) {
149 cholmod_dense *result = new cholmod_dense;
150 result->nrow = vector.size();
151 result->ncol = 1;
152 result->nzmax = vector.size();
153 result->d = vector.size();
154 result->x = const_cast<double *>(vector.data());
155 result->z = nullptr;
156 result->xtype = CHOLMOD_REAL;
157 result->dtype = CHOLMOD_DOUBLE;
158 return result;
159 }
160
161 static Eigen::VectorXd viewAsEigen(cholmod_dense *vectorPtr) {
162 return Eigen::VectorXd::Map(reinterpret_cast<double *>(vectorPtr->x), vectorPtr->nrow);
163 }
