57 paths.pop_front(); // Remove the oldest path

72 void computeRC() {
73 for (auto &path : paths) {
74 int iteration_added = std::get<0>(path); // Get the iteration when the path was added
75
76 // Stop processing if the path is older than current_iteration + max_life
77 if (iteration_added + max_live_time < current_iteration) { break; }
78
79 Path &p = std::get<1>(path);
80 p.red_cost = p.cost;
81
82 if (p.size() > 3) {
83 for (int i = 1; i < p.size() - 1; i++) {
84 auto &node = (*nodes)[p[i]]; // Dereference nodes and access element
85 p.red_cost -= node.cost;
86 }
87 }
88 }
89 }

91 std::vector<Path> get_paths_with_negative_red_cost() {
92 std::vector<Path> result;
93
94 // Remove the paths that are older than current_iteration + max_life or have a negative red_cost
95 auto it = paths.begin();
96 while (it != paths.end()) {
97 int iteration_added = std::get<0>(*it); // Get the iteration when the path was added
98 const Path &p = std::get<1>(*it);
99
100 // If the path is older than max_live_time, or has a negative red_cost
101 if (iteration_added + max_live_time < current_iteration || p.red_cost < 0) {
102 if (p.red_cost < 0) {
103 result.push_back(p); // Add paths with negative red_cost to the result
104 }
105 it = paths.erase(it); // Remove from paths and move iterator to the next element
106 } else {
107 ++it; // Only move the iterator if not erasing
108 }
109 }
110
111 // Sort the result based on red_cost
112 std::sort(result.begin(), result.end(), [](const Path &a, const Path &b) { return a.red_cost < b.red_cost; });
113
114 return result;
115 }

136public:
137 explicit LabelPool(size_t initial_pool_size, size_t max_pool_size = 5000000)
138 : pool_size(initial_pool_size), max_pool_size(max_pool_size),
139 pool(std::pmr::get_default_resource()) // Default memory resource
140 {
141 allocate_labels(pool_size);
142 }
143
144 Label *acquire() {
145 if (!available_labels.empty()) {
146 Label *new_label = available_labels.back();
147 available_labels.pop_back();
148 in_use_labels.push_back(new_label);
149 new_label->reset();
150 return new_label;
151 }
152
153 // Allocate label using the pool resource
154 auto *new_label = new (pool.allocate(sizeof(Label))) Label();
155 in_use_labels.push_back(new_label);
156 return new_label;
157 }
158
159 void reset() {
160 available_labels.reserve(available_labels.size() + in_use_labels.size());
161
162 available_labels.insert(available_labels.end(), std::make_move_iterator(in_use_labels.begin()),
163 std::make_move_iterator(in_use_labels.end()));
164
165 in_use_labels.clear();
166 }
167
168private:
169 void allocate_labels(size_t count) {
170 available_labels.reserve(count);
171 for (size_t i = 0; i < count; ++i) { available_labels.push_back(new (pool.allocate(sizeof(Label))) Label()); }
172 }
173
174 void cleanup() {
175 for (auto &label : available_labels) {
176 label->~Label();
177 pool.deallocate(label, sizeof(Label));
178 }
179 available_labels.clear();
180
181 for (auto &label : in_use_labels) {
182 label->~Label();
183 pool.deallocate(label, sizeof(Label));
184 }
185 in_use_labels.clear();
186
187 for (auto &label : deleted_states) {
188 label->~Label();
189 pool.deallocate(label, sizeof(Label));
190 }
191 deleted_states.clear();
192 }
193
194 size_t pool_size;
195 size_t max_pool_size;
196
197 std::pmr::unsynchronized_pool_resource pool; // Memory pool
198 std::pmr::vector<Label *> available_labels{&pool}; // Labels ready to be acquired
199 std::pmr::vector<Label *> in_use_labels{&pool}; // Labels currently in use
200 std::vector<Label *> deleted_states; // Labels available for recycling
201};
