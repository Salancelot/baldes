32 // Bisect the matrix into supernodes
33 Supernode *bisect(const Eigen::SparseMatrix<double> &matrix, int row_start = 0, int threshold = 64) {
34 int n = matrix.rows();
35
36 // Step 1: Reorder the matrix using custom AMD
37 std::vector<int> amd_ordering = custom_amd(matrix);
38 // convert amd ordering to Eigen permutation
39 Eigen::PermutationMatrix<Eigen::Dynamic> perm(n);
40 for (int i = 0; i < n; ++i) { perm.indices()[i] = amd_ordering[i]; }
41
42 // Step 2: Permute the matrix based on AMD ordering
43 // Eigen::PermutationMatrix<Eigen::Dynamic> perm(amd_ordering);
44 Eigen::SparseMatrix<double> permuted_matrix = perm.transpose() * matrix * perm;
45
46 // Step 3: Recursively bisect the permuted matrix
47 if (n <= threshold) {
48 // Base case: create a leaf supernode
49 std::vector<int> structure;
50 for (int k = 0; k < permuted_matrix.outerSize(); ++k) {
51 if (permuted_matrix.col(k).nonZeros() > 0) { structure.push_back(k); }
52 }
53 return new Supernode(row_start, n, structure);
54 }
55
56 int mid = n / 2;
57
58 Eigen::SparseMatrix<double> top_left = extract_sparse_block(permuted_matrix, 0, 0, mid, mid);
59 Eigen::SparseMatrix<double> bottom_right = extract_sparse_block(permuted_matrix, mid, mid, n - mid, n - mid);
60
61 Supernode *left_supernode = bisect(top_left, row_start, threshold);
62 Supernode *right_supernode = bisect(bottom_right, row_start + mid, threshold);
63
64 // Create the parent node, accounting for the entire structure
65 std::vector<int> parent_structure;
66 for (int k = 0; k < permuted_matrix.outerSize(); ++k) {
67 if (permuted_matrix.col(k).nonZeros() > 0) { parent_structure.push_back(k); }
68 }
69
70 Supernode *parent_supernode = new Supernode(row_start, n, parent_structure);
71 parent_supernode->add_child(left_supernode);
72 parent_supernode->add_child(right_supernode);
73
74 return parent_supernode;
75 }

93 std::vector<int> custom_amd(const Eigen::SparseMatrix<double> &matrix) {
94 int n = matrix.rows();
95
96 // Create a graph representation where each row/column corresponds to a node
97 std::vector<std::set<int>> adjacency_list(n);
98
99 // Populate the adjacency list based on non-zero entries in the matrix
100 for (int k = 0; k < matrix.outerSize(); ++k) {
101 for (Eigen::SparseMatrix<double>::InnerIterator it(matrix, k); it; ++it) {
102 if (it.row() != it.col()) {
103 adjacency_list[it.row()].insert(it.col());
104 adjacency_list[it.col()].insert(it.row());
105 }
106 }
107 }
108
109 // Vector to store the final ordering
110 std::vector<int> ordering;
111 std::set<std::pair<int, int>> degree_queue; // (degree, node)
112
113 // Initialize the degrees (number of non-zero connections) and priority queue
114 std::vector<int> degrees(n);
115 for (int i = 0; i < n; ++i) {
116 degrees[i] = adjacency_list[i].size();
117 degree_queue.insert({degrees[i], i});
118 }
119
120 while (!degree_queue.empty()) {
121 // Choose the node with the lowest degree
122 int node = degree_queue.begin()->second;
123 degree_queue.erase(degree_queue.begin());
124
125 // Add this node to the ordering
126 ordering.push_back(node);
127
128 // Update the graph by connecting the neighbors of the chosen node
129 std::set<int> neighbors = adjacency_list[node];
130 for (int neighbor : neighbors) {
131 adjacency_list[neighbor].erase(node);
132 for (int other_neighbor : neighbors) {
133 if (neighbor != other_neighbor) { adjacency_list[neighbor].insert(other_neighbor); }
134 }
135 // Update the degrees of the remaining neighbors
136 degree_queue.erase({degrees[neighbor], neighbor});
137 degrees[neighbor] = adjacency_list[neighbor].size();
138 degree_queue.insert({degrees[neighbor], neighbor});
139 }
140 }
141
142 return ordering;
143 }

145 std::vector<int> symbolic_factorization(Supernode *supernode) {
146 if (supernode->children.empty()) {
147 // Leaf node: return its own structure
148 return supernode->structure;
149 }
150
151 // Collect structure from children
152 std::vector<int> struct_union = supernode->structure;
153
154 for (auto *child : supernode->children) {
155 std::vector<int> child_structure = symbolic_factorization(child);
156 struct_union.insert(struct_union.end(), child_structure.begin(), child_structure.end());
157 }
158
159 // Ensure unique structure (can use set for uniqueness)
160 std::sort(struct_union.begin(), struct_union.end());
161 struct_union.erase(std::unique(struct_union.begin(), struct_union.end()), struct_union.end());
162
163 supernode->structure = struct_union;
164 return struct_union;

176 void numeric_factorization(Supernode *supernode, const Eigen::SparseMatrix<double> &matrix,
177 double regularizationFactor = 1e-5, double eps = 1e-8, double delta = 1e-3) {
178 if (supernode->children.empty()) {
179 Eigen::SparseMatrix<double> block =
180 extract_sparse_block(matrix, supernode->start, supernode->start, supernode->size, supernode->size);
181
182 if (block.nonZeros() > supernode->size * supernode->size * 0.5) {
183 // Dense case
184 Eigen::MatrixXd dense_block = Eigen::MatrixXd(block); // Convert to dense
185 Eigen::LDLT<Eigen::MatrixXd> ldlt;
186 ldlt.compute(dense_block);
187
188 if (ldlt.info() != Eigen::Success) {
189 std::cerr << "Dense LDLT decomposition failed. Applying regularization..." << std::endl;
190 apply_aggressive_regularization(dense_block, regularizationFactor);
191 ldlt.compute(dense_block);
192
193 if (ldlt.info() != Eigen::Success) {
194 std::cerr << "Dense LDLT decomposition still failed after regularization." << std::endl;
195 return;
196 }
197 }
198
199 // Dynamic regularization based on diagonal values
200 Eigen::VectorXd D = ldlt.vectorD(); // Diagonal
201 for (int i = 0; i < D.size(); ++i) {
202 if (D(i) < eps) {
203 D(i) = delta; // Apply regularization if too small
204 }
205 }
206
207 // Convert dense triangular matrix to sparse
208 Eigen::MatrixXd dense_L = ldlt.matrixL(); // Extract dense L
209 supernode->L = dense_L.sparseView(); // Convert to sparse matrix
210 supernode->D = D; // Set regularized diagonal
211 supernode->P.setIdentity(supernode->size);
212
213 } else {
214 // Sparse case
215 Eigen::SimplicialLDLT<Eigen::SparseMatrix<double>> ldlt;
216 ldlt.compute(block);
217
218 if (ldlt.info() != Eigen::Success) {
219 std::cerr << "Sparse LDLT decomposition failed. Applying regularization..." << std::endl;
220 apply_aggressive_regularization(block, regularizationFactor);
221 ldlt.compute(block);
222
223 if (ldlt.info() != Eigen::Success) {
224 std::cerr << "Sparse LDLT decomposition still failed after regularization." << std::endl;
225 return;
226 }
227 }
228
229 // Apply dynamic regularization to diagonal values
230 Eigen::VectorXd D = ldlt.vectorD();
231 for (int i = 0; i < D.size(); ++i) {
232 if (std::abs(D(i)) < eps) {
233 D(i) = (D(i) < 0) ? -delta : delta; // Apply delta with correct sign
234 }
235 }
236
237 supernode->L = ldlt.matrixL();
238 supernode->D = D; // Set regularized diagonal
239 supernode->P = Eigen::PermutationMatrix<Eigen::Dynamic>(ldlt.permutationP());
240 }
241
242 return;
243 }
244
245 // Recursively process children supernodes
246 for (auto *child : supernode->children) {
247 numeric_factorization(child, matrix, regularizationFactor, eps, delta);
248 }
249
250 // Process non-leaf supernodes
251 Eigen::SparseMatrix<double> block =
252 extract_sparse_block(matrix, supernode->start, supernode->start, supernode->size, supernode->size);
253 Eigen::SimplicialLDLT<Eigen::SparseMatrix<double>> ldlt;
254 ldlt.compute(block);
255
256 if (ldlt.info() != Eigen::Success) {
257 std::cerr << "LDLT decomposition failed. Applying regularization..." << std::endl;
258 apply_regularization(block, regularizationFactor);
259 ldlt.compute(block);
260
261 if (ldlt.info() != Eigen::Success) {
262 std::cerr << "LDLT decomposition still failed after regularization." << std::endl;
263 return;
264 }
265 }
266
267 // Apply dynamic regularization
268 Eigen::VectorXd D = ldlt.vectorD();
269 for (int i = 0; i < D.size(); ++i) {
270 if (std::abs(D(i)) < eps) { D(i) = (D(i) < 0) ? -delta : delta; }
271 }
272
273 supernode->L = ldlt.matrixL();
274 supernode->D = D;
275 supernode->P = Eigen::PermutationMatrix<Eigen::Dynamic>(ldlt.permutationP());

277 void apply_aggressive_regularization(Eigen::MatrixXd &matrix, double &regularization, double threshold = 1e-6) {
278 const double max_regularization = 1e-3;
279 const double regularization_increment = 10;
280
281 while (regularization < max_regularization) {
282 bool regularized = false;
283 for (int i = 0; i < matrix.rows(); ++i) {
284 if (std::abs(matrix(i, i)) < threshold) {
285 matrix(i, i) += regularization;
286 regularized = true;
287 }
288 }
289 if (regularized) break;
290 regularization *= regularization_increment;
291 }

313 Eigen::SimplicialLDLT<Eigen::SparseMatrix<double>> &ldlt,
314 double &regularizationFactor, int max_retries = 5) {
315 int retries = 0;
316 while (retries < max_retries) {
317 bool has_nan = false;
318 for (int k = 0; k < block.outerSize(); ++k) {
319 for (Eigen::SparseMatrix<double>::InnerIterator it(block, k); it; ++it) {
320 if (std::isnan(it.value())) {

338 retries++;
