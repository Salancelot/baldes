96// Hash function for RawArc
97struct RawArcHash {
98 std::size_t operator()(const RawArc &arc) const {
99 // Take the address of 'arc.from' and 'arc.to' for hashing
100 XXH64_hash_t hash_value = XXH3_64bits(&arc.from, sizeof(arc.from)); // Hash the 'from' field
101 hash_value = XXH3_64bits_withSeed(&arc.to, sizeof(arc.to), hash_value); // Combine with 'to' field
102
103 return static_cast<std::size_t>(hash_value); // Return the final hash value
104 }
105};
