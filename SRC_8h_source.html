242template <typename T>
243inline void combinations(const std::vector<T> &elements, int k, int max_combinations,
244 std::vector<std::vector<T>> &result) {
245 std::vector<int> indices(k);
246 std::iota(indices.begin(), indices.end(), 0);
247
248 // Random number generator
249 Xoroshiro128Plus rng; // Seed it (you can change the seed)
250 std::uniform_real_distribution<> dis(0.0, 1.0);
251
252 int total_combinations = 0; // Track total combinations generated
253
254 while (true) {
255 // Generate the current combination based on the selected indices
256 std::vector<T> combination;
257 for (int idx : indices) { combination.push_back(elements[idx]); }
258
259 // Randomly select this combination
260 ++total_combinations;
261 if (result.size() < max_combinations || dis(rng) < static_cast<double>(max_combinations) / total_combinations) {
262 if (result.size() < max_combinations) {
263 result.push_back(combination);
264 } else {
265 // Replace a random element in the result if the result is already full
266 std::uniform_int_distribution<> idx_dis(0, max_combinations - 1);
267 result[idx_dis(rng)] = combination;
268 }
269 }
270
271 // Find the rightmost index that can be incremented
272 int i = k - 1;
273 while (i >= 0 && indices[i] == elements.size() - k + i) { --i; }
274
275 // If no such index exists, all combinations are generated
276 if (i < 0) break;
277
278 // Increment this index
279 ++indices[i];
280
281 // Update the following indices
282 for (int j = i + 1; j < k; ++j) { indices[j] = indices[j - 1] + 1; }
283 }
284}

287inline uint64_t hashVector(const std::vector<int> &vec) {
288 uint64_t hash = 0; // Initialize seed for XXHash
289
290 for (const int &elem : vec) {
291 // Combine each element's hash using XXHash and XOR it with the running hash value
292 hash ^= XXH3_64bits_withSeed(&elem, sizeof(elem), hash);
293 }
294
295 return hash;
296}

298// Hash function for a vector of doubles
299inline uint64_t hashVector(const std::vector<double> &vec) {
300 uint64_t hash = 0; // Initialize seed for XXHash
301
302 for (const double &elem : vec) {
303 // Convert the double to uint64_t for hashing
304 std::uint64_t bit_rep = std::bit_cast<std::uint64_t>(elem);
305 // Combine each element's hash using XXHash and XOR it with the running hash value
306 hash ^= XXH3_64bits_withSeed(&bit_rep, sizeof(bit_rep), hash);
307 }
308
309 return hash;
310}
