191 void add_sorted_label(Label *label) noexcept {
192 if (labels_vec.empty() || label->cost >= labels_vec.back()->cost) {
193 labels_vec.push_back(label); // Direct insertion at the end
194 } else if (label->cost <= labels_vec.front()->cost) {
195 labels_vec.insert(labels_vec.begin(), label); // Direct insertion at the beginning
196 } else {
197 auto it = std::lower_bound(labels_vec.begin(), labels_vec.end(), label,
198 [](const Label *a, const Label *b) { return a->cost < b->cost; });
199 labels_vec.insert(it, label); // Insertion in the middle
200 }
201 }

212 void add_label_lim(Label *label, size_t limit) noexcept {
213 if (labels_vec.size() < limit) {
214 labels_vec.push_back(label);

216 auto it = std::max_element(labels_vec.begin(), labels_vec.end(),
217 [](const Label *a, const Label *b) { return a->cost > b->cost; });
218 if (label->cost < (*it)->cost) { *it = label; }
219 }
