246 DualSolution getStabDualSolAdvanced(const DualSolution &input_duals) {
247 std::vector<double> nodeDuals;
248 nodeDuals.assign(input_duals.begin(), input_duals.begin() + sizeDual);
249 // If there's no stabilization center, return the input duals
250 if (cur_stab_center.empty()) { return nodeDuals; }
251
252 // Initialize subgradient if it hasn't been initialized yet
253 if (subgradient.empty()) {
254 subgradient.assign(nodeDuals.size(), 0.0); // Placeholder: Replace with real subgradient initialization
255 }
256
257 // Initialize the separation duals and smooth dual solution if they are empty
258 if (duals_sep.empty()) {
259 duals_sep = nodeDuals; // Initialize with the node duals
260 smooth_dual_sol = duals_sep; // Also initialize smooth_dual_sol with the same
261 return nodeDuals; // Return the input duals for now
262 }
263
264 // Get the current stabilization center (duals_in) and the current node duals (duals_out)
265 const auto &duals_in = cur_stab_center;
266 const auto &duals_out = nodeDuals;
267
268 const size_t n = nodeDuals.size();
269 std::vector<double> duals_tilde(n, 0.0);
270 std::vector<double> duals_g(n, 0.0);
271 std::vector<double> rho(n, 0.0);
272
273 // Precompute norms that are used multiple times
274 double norm_in_out = norm(duals_in, duals_out); // Norm of (duals_out - duals_in)
275 double norm_subgradient = norm(subgradient); // Norm of the subgradient
276
277 for (size_t row_id = 0; row_id < n; ++row_id) {
278 duals_tilde[row_id] = cur_alpha * duals_in[row_id] + (1 - cur_alpha) * duals_out[row_id];
279 // duals_tilde[row_id] = cur_alpha * cur_stab_center[row_id] + (1 - cur_alpha) * nodeDuals[row_id];
280 }
281 // Compute the coefficient for π_g based on the norm of duals_in and duals_out
282 double coef_g = norm(duals_in, duals_out) / norm(subgradient);
283
284 // Compute π_g: duals_in + coef_g * subgradient
285 for (size_t row_id = 0; row_id < n; ++row_id) {
286 duals_g[row_id] = duals_in[row_id] + coef_g * subgradient[row_id];
287 }
288
289 if (nb_misprices > 0) {
290 beta = 0.0;
291 } else {
292 double dot_product = 0;
293 for (auto row_id = 0; row_id < duals_out.size(); ++row_id) {
294 dot_product += (duals_out[row_id] - duals_in[row_id]) * (duals_g[row_id] - duals_in[row_id]);
295 }
296 beta = dot_product / (norm(duals_in, duals_out) * norm(duals_in, duals_g));
297 }
298 beta = std::max(0.0, beta);
299 // fmt::print("beta: {}\n", beta);
300
301 // Compute ρ: a combination of duals_g and duals_out based on β
302 for (size_t row_id = 0; row_id < n; ++row_id) {
303 rho[row_id] = beta * duals_g[row_id] + (1 - beta) * duals_out[row_id]; // Weighted sum
304 }
305 double coef_sep = norm(duals_in, duals_tilde) / norm(duals_in, rho);
306
307 // Update the duals_sep by adjusting duals_in towards ρ
308 for (size_t row_id = 0; row_id < n; ++row_id) {
309 duals_sep[row_id] = duals_in[row_id] + coef_sep * (rho[row_id] - duals_in[row_id]);
310 }
311
312 // Update the smooth dual solution and return it
313 smooth_dual_sol = duals_sep;
314 return duals_sep;
315 }

329 bool dynamic_alpha_schedule(const ModelData &dados) {
330
331 // Get the number of rows from the size of the nodeDuals
332 size_t number_of_rows = cur_stab_center.size();

334 std::vector<double> in_sep_direction(number_of_rows, 0.0);
335 in_sep_direction = sub(smooth_dual_sol, cur_stab_center);

338
339 if (in_sep_dir_norm == 0) { return false; }
340
341 // Compute dot product of in_sep_direction and subgradient
342 double dot_product = 0.0;
343 for (size_t row_id = 0; row_id < number_of_rows; ++row_id) {
344 dot_product += subgradient[row_id] * in_sep_direction[row_id];
345 }
346 double cos_angle = dot_product / (in_sep_dir_norm * subgradient_norm);
347
348 return cos_angle < 1e-12;

351 void update_subgradient(const ModelData &dados, const DualSolution &nodeDuals,
352 const std::vector<Label *> &best_pricing_cols) {
353 size_t number_of_rows = nodeDuals.size();
354
355 // Initialize new_rows and apply contributions from best_pricing_cols, skipping positive-cost columns
356 new_rows.assign(number_of_rows, 0.0);
357
358 // Iterate through the columns brought by the pricing step
359 for (const auto *best_pricing_col : best_pricing_cols) {
360 // Skip columns with positive cost, as they won't improve the solution
361 if (best_pricing_col->cost >= 0) { continue; }
362
363 // Update rows based on the nodes covered by this column
364 for (const auto &row : best_pricing_col->nodes_covered) {
365 // Ensure valid row indices (ignore invalid rows and the last node)
366 if (row > 0 && row != N_SIZE - 1) {
367 new_rows[row - 1] += 1; // Apply contribution of the column to the row
368 }
369 }
370 }
371
372 // Define and set row bounds based on the problem constraints (dados.sense)
373 // Reuse vectors to avoid reallocations (optimize performance)
374 static std::vector<double> new_row_lower_bounds, new_row_upper_bounds;
375
376 // Resize bounds vectors if needed
377 if (new_row_lower_bounds.size() != number_of_rows) {
378 new_row_lower_bounds.resize(number_of_rows);
379 new_row_upper_bounds.resize(number_of_rows);
380 }
381
382 // Set row bounds based on the constraint sense ('<', '>', '=')
383 for (size_t row_id = 0; row_id < number_of_rows; ++row_id) {
384 // Check the constraint type in dados.sense
385 if (dados.sense[row_id] == '<') {
386 // ax < b
387 // ax - b < 0
388 // b - ax > 0
389 // If sense is '<', set upper bound and negative infinity lower bound
390 new_row_upper_bounds[row_id] = dados.b[row_id];
391 new_row_lower_bounds[row_id] = -std::numeric_limits<double>::infinity();
392 } else if (dados.sense[row_id] == '>') {
393 // ax > b
394 // ax - b > 0
395 // b - ax < 0
396 // If sense is '>', set lower bound and positive infinity upper bound
397 new_row_upper_bounds[row_id] = std::numeric_limits<double>::infinity();
398 new_row_lower_bounds[row_id] = dados.b[row_id];
399 } else {
400 // If sense is '=', set both lower and upper bounds to be equal to dados.b[row_id]
401 new_row_upper_bounds[row_id] = dados.b[row_id];
402 new_row_lower_bounds[row_id] = dados.b[row_id];
403 }
404 }
405
406 // Update the subgradient based on new row values and bounds
407 subgradient.assign(number_of_rows, 0.0);
408
409 for (size_t row_id = 0; row_id < number_of_rows; ++row_id) {
410 // subgradient[row_id] = std::min(0.0, new_row_upper_bounds[row_id] - new_rows[row_id]) +
411 // std::max(0.0, new_row_lower_bounds[row_id] - new_rows[row_id]);
412 if (dados.sense[row_id] == '<') {
413 subgradient[row_id] = new_rows[row_id] - new_row_upper_bounds[row_id];
414 } else if (dados.sense[row_id] == '>') {
415 subgradient[row_id] = new_row_lower_bounds[row_id] - new_rows[row_id];
416 } else if (dados.sense[row_id] == '=') {
417 subgradient[row_id] = new_row_upper_bounds[row_id] - new_rows[row_id];
418 }
419 }
420 subgradient_norm = norm(subgradient);
421 }

450 void update_stabilization_after_pricing_optim(const ModelData &dados, const DualSolution &input_duals,
451 const double &lag_gap, std::vector<Label *> best_pricing_cols) {
452 std::vector<double> nodeDuals;
453 nodeDuals.assign(input_duals.begin(), input_duals.begin() + sizeDual);
454 if (nb_misprices == 0) {
455 update_subgradient(dados, nodeDuals, best_pricing_cols);
456 auto alpha_dir = dynamic_alpha_schedule(dados);
457 if (alpha_dir) {
458 alpha = std::min(0.99, base_alpha + (1.0 - base_alpha) * 0.1); // Increase cur_alpha
459 } else {
460 alpha = std::max(0.0, base_alpha - 0.1); // Decrease cur_alpha
461 }
462 }
463 if (lag_gap < lag_gap_prev) {
464 stab_center_for_next_iteration = smooth_dual_sol;
465 } else {
466 stab_center_for_next_iteration = cur_stab_center;
467 }
468 lag_gap_prev = lag_gap;
469 base_alpha = alpha;
470 }
