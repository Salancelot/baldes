170 inline constexpr bool process_all_resources(std::vector<double> &new_resources,
171 const std::array<double, R_SIZE> &initial_resources, const Gamma &gamma,
172 const VRPNode &theNode) {
173 if constexpr (I < N) {
174 // Process the resource at index I
175 if (!process_resource<D, I, N>(new_resources[I], initial_resources, gamma, theNode)) {
176 return false; // Constraint violated, return false
177 }
178 // Recur to process the next resource
179 return process_all_resources<D, I + 1, N>(new_resources, initial_resources, gamma, theNode);
180 }
181 return true; // All resources processed successfully
182 }

194 inline constexpr bool process_resource(double &new_resource, const std::array<double, R_SIZE> &initial_resources,
195 const Gamma &gamma, const VRPNode &theNode) {
196 if constexpr (resource_disposability[I] == 1) { // Checked at compile-time
197 if constexpr (D == Direction::Forward) {
198 new_resource =
199 std::max(initial_resources[I] + gamma.resource_increment[I], static_cast<double>(theNode.lb[I]));
200 if (new_resource > theNode.ub[I]) {
201 return false; // Exceeds upper bound, return false to stop processing
202 }
203 } else {
204 new_resource =
205 std::min(initial_resources[I] - gamma.resource_increment[I], static_cast<double>(theNode.ub[I]));
206 if (new_resource < theNode.lb[I]) {
207 return false; // Below lower bound, return false to stop processing
208 }
209 }
210 } else if constexpr (resource_disposability[I] == 0) {
211 // TODO: Non-disposable resource handling, check if it is right
212 if constexpr (D == Direction::Forward) {
213 new_resource = initial_resources[I] + gamma.resource_increment[I];
214 if (new_resource > theNode.ub[I]) {
215 return false; // Exceeds upper bound, return false to stop processing
216 } else if (new_resource < theNode.lb[I]) {
217 return false; // Below lower bound, return false to stop processing
218 }
219 } else {
220 new_resource = initial_resources[I] - gamma.resource_increment[I];
221 if (new_resource > theNode.ub[I]) {
222 return false; // Exceeds upper bound, return false to stop processing
223 } else if (new_resource < theNode.lb[I]) {
224 return false; // Below lower bound, return false to stop processing
225 }
226 }
227 } else if constexpr (resource_disposability[I] == 2) {
228 // TODO:: Binary resource handling, check if logic is right
229 if constexpr (D == Direction::Forward) {
230 // For binary resources, flip between 0 and 1 based on gamma.resource_increment[I]
231 if (gamma.resource_increment[I] > 0) {
232 new_resource = 1.0; // Switch "on"
233 } else {
234 new_resource = 0.0; // Switch "off"
235 }
236 } else {
237 // In reverse, toggle as well
238 if (gamma.resource_increment[I] > 0) {
239 new_resource = 0.0; // Reverse logic: turn "off"
240 } else {
241 new_resource = 1.0; // Reverse logic: turn "on"
242 }
243 }
244 } else if constexpr (resource_disposability[I] == 3) {
245 // TODO: handling multiple time windows case
246 // "OR" resource case using mtw_lb and mtw_ub vectors for multiple time windows
247 if constexpr (D == Direction::Forward) {
248 bool is_feasible = false;
249 for (size_t i = 0; i < theNode.mtw_lb.size(); ++i) {
250 new_resource = std::max(initial_resources[I] + gamma.resource_increment[I], theNode.mtw_lb[i]);
251 if (new_resource > theNode.ub[I]) {
252 continue; // Exceeds upper bound, try next time window
253 } else {
254 is_feasible = true; // Feasible in this time window
255 break;
256 }
257 }
258
259 if (!is_feasible) {
260 return false; // Not feasible in any of the ranges
261 }
262
263 return true; // Successfully processed all resources
264 } else {
265 bool is_feasible = false;
266 for (size_t i = 0; i < theNode.mtw_ub.size(); ++i) {
267 new_resource = std::min(initial_resources[I] - gamma.resource_increment[I], theNode.mtw_ub[i]);
268 if (new_resource < theNode.lb[I]) {
269 continue; // Below lower bound, try next time window }
270 } else {
271 is_feasible = true; // Feasible in this time window break;
272 }
273 }
274
275 if (!is_feasible) {
276 return false; // Not feasible in any of the ranges
277 }
278 }
279 }
280 return true; // Successfully processed all resources

379 void define_elementarity_sets() {
380 size_t num_nodes = nodes.size();
381
382 // Initialize elementarity sets
383 elementarity_bitmap.resize(num_nodes); // Bitmap for elementarity sets
384
385 for (size_t i = 0; i < num_nodes; ++i) {
386 // Each customer should appear in its own elementarity set
387 size_t num_segments = (num_nodes + 63) / 64;
388 elementarity_bitmap[i].resize(num_segments, 0);
389
390 // Mark the node itself in the elementarity set
391 size_t segment_self = i >> 6;
392 size_t bit_position_self = i & 63;
393 elementarity_bitmap[i][segment_self] |= (1ULL << bit_position_self);
394 }

739 if constexpr (D == Direction::Forward) {
740 for (int r = 0; r < bucket.lb.size(); ++r) { mid_point[r] = (bucket.lb[r] + base_intervals[r]) / 2.0; }
741 } else {
742 for (int r = 0; r < bucket.lb.size(); ++r) { mid_point[r] = (bucket.ub[r] - base_intervals[r]) / 2.0; }
743 }
744
745 // Create two new buckets by splitting at the mid-point
746 Bucket bucket1, bucket2;
747 if constexpr (D == Direction::Forward) {
748 bucket1 = Bucket(bucket.node_id, bucket.lb, bucket.ub);
749 bucket2 = Bucket(bucket.node_id, mid_point, bucket.ub);
750 } else {
751 bucket1 = Bucket(bucket.node_id, bucket.lb, bucket.ub);
752 bucket2 = Bucket(bucket.node_id, bucket.lb, mid_point);
753 }
754
755 // Manually shift elements to the right to make space for the new bucket
756 for (int j = original_num_buckets; j > i + 1; --j) { buckets[j] = buckets[j - 1]; }
757
758 // Insert the new bucket at position i+1
759 buckets[i + 1] = bucket2;
760
761 // Replace the current bucket with the first half (bucket1)
762 buckets[i] = bucket1;
763
764 // Track the number of splits for this node_id
765 splits_per_node[bucket.node_id]++;
766
767 // Since we added a new bucket at i+1, increment i to skip over the new bucket
768 ++i;
769 ++original_num_buckets; // Adjust the count to account for the new bucket
770 }
771
772 // Update the global bucket size
773 buckets_size = original_num_buckets;
774
775 // Now update the num_buckets_index for each node_id based on the final positions
776 int current_index = 0;
777 for (int i = 0; i < original_num_buckets; ++i) {
778 const auto &bucket = buckets[i];
779 if (num_buckets_index[bucket.node_id] == -1) { num_buckets_index[bucket.node_id] = current_index; }
780 current_index++;
781 }
782 }

844 inline bool check_feasibility(const Label *fw_label, const Label *bw_label) {
845 if (!fw_label || !bw_label) return false;
846
847 // Cache resources and node data
848 const auto &fw_resources = fw_label->resources;
849 const auto &bw_resources = bw_label->resources;
850 const struct VRPNode &VRPNode = nodes[fw_label->node_id];
851
852 // Time feasibility check
853 const auto time_fw = fw_resources[TIME_INDEX];
854 const auto time_bw = bw_resources[TIME_INDEX];
855 const auto travel_time = getcij(fw_label->node_id, bw_label->node_id);
856 if (time_fw + travel_time + VRPNode.duration > time_bw) { return false; }
857
858 // Resource feasibility check (if applicable)
859 if constexpr (R_SIZE > 1) {
860 for (size_t i = 1; i < R_SIZE; ++i) {
861 const auto resource_fw = fw_resources[i];
862 const auto resource_bw = bw_resources[i];
863 const auto demand = VRPNode.demand;
864
865 if (resource_fw + demand > resource_bw) { return false; }
866 }
867 }
868
869 return true;
870 }

874 // ADAPTIVE TERMINAL TIME
876 // Adjust the terminal time dynamically based on the difference between the number of forward and
877 // backward labels
878 for (auto &split : q_star) {
879 // print n_fw_labels and n_bw_labels
880 // If there are more backward labels than forward labels, increase the terminal time slightly
881 if (((static_cast<double>(n_bw_labels) - static_cast<double>(n_fw_labels)) /
882 static_cast<double>(n_fw_labels)) > 0.05) {
883 split += 0.05 * R_max[TIME_INDEX];
884 }
885 // If there are more forward labels than backward labels, decrease the terminal time slightly
886 else if (((static_cast<double>(n_fw_labels) - static_cast<double>(n_bw_labels)) /
887 static_cast<double>(n_bw_labels)) > 0.05) {
888 split -= 0.05 * R_max[TIME_INDEX];
889 }
890 }

895 if (bucket_interval > 100) { return false; }
896 // compute the mean of dominance_checks_per_bucket
897 double mean_dominance_checks = 0.0;
898 for (size_t i = 0; i < dominance_checks_per_bucket.size(); ++i) {
899 mean_dominance_checks += dominance_checks_per_bucket[i];
900 }
901 auto step_calc = mean_dominance_checks / non_dominated_labels_per_bucket;
902 if (step_calc > 1000) {
903

905 print_info("Increasing bucket interval to {}\n", bucket_interval * 2);
906 bucket_interval = bucket_interval * 2;
907 redefine(bucket_interval);
908 updated = true;
909 fixed = false;
910 }
