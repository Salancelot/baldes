156 if (s1) {
157 stage = 1;
158 paths = bi_labeling_algorithm<Stage::One>(); // Solve the problem with Stage 1 heuristic
159 inner_obj = paths[0]->cost;
160
161 // Transition from Stage 1 to Stage 2 if the objective improves or after 10 iterations

163 s1 = false;

187template <Direction D, Stage S, Full F>
188std::vector<double> BucketGraph::labeling_algorithm() noexcept {
189
190 // Assign the correct direction buckets, ordered SCCs, and other related structures depending on the direction D
191 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
192 auto &ordered_sccs = assign_buckets<D>(fw_ordered_sccs, bw_ordered_sccs);
193 auto &topological_order = assign_buckets<D>(fw_topological_order, bw_topological_order);
194 auto &sccs = assign_buckets<D>(fw_sccs, bw_sccs);
195 auto &Phi = assign_buckets<D>(Phi_fw, Phi_bw); // Forward/backward adjacency list of strongly connected components
196 auto &c_bar = assign_buckets<D>(fw_c_bar, bw_c_bar); // Lower bound on the cost of labels in each bucket
197 auto &fixed_buckets = assign_buckets<D>(fw_fixed_buckets, bw_fixed_buckets);
198 auto &n_labels = assign_buckets<D>(n_fw_labels, n_bw_labels); // Number of labels processed
199 auto &sorted_sccs = assign_buckets<D>(fw_sccs_sorted, bw_sccs_sorted); // Sorted SCCs
200 auto &n_buckets = assign_buckets<D>(fw_buckets_size, bw_buckets_size); // Total number of buckets
201 auto &stat_n_labels = assign_buckets<D>(stat_n_labels_fw, stat_n_labels_bw);
202 auto &stat_n_dom = assign_buckets<D>(stat_n_dom_fw, stat_n_dom_bw);
203
204 // Reset the number of labels processed for this run
205 n_labels = 0;
206
207 // Set up a vector for tracking visited buckets, each segment represents 64 buckets
208 const size_t n_segments = n_buckets / 64 + 1;
209 std::vector<uint64_t> Bvisited(n_segments, 0);
210
211 bool all_ext; // Flag to indicate if all labels have been extended
212 bool dominated; // Flag to check if a label is dominated
213 bool domin_smaller; // Flag to check if a label is dominated by a smaller bucket
214
215 auto a_ctr = 0;
216 // Iterate through each strongly connected component (SCC) in topological order
217 for (const auto &scc_index : topological_order) {
218 do {
219 all_ext = true; // Assume all labels have been extended at the start
220 for (const auto bucket : sorted_sccs[scc_index]) {
221 auto &bucket_labels = buckets[bucket].get_labels(); // Get unextended labels from this bucket
222 if (bucket_labels.empty()) { continue; } // Skip empty buckets
223 for (Label *label : bucket_labels) {
224 if (label->is_extended) { continue; } // Skip labels that have already been extended
225
226 // NOTE: double check if this is the best way to handle this
227 if constexpr (F == Full::Partial) {
228 if constexpr (D == Direction::Forward) {
229 if (label->resources[TIME_INDEX] > q_star[TIME_INDEX]) {
230 label->set_extended(true);
231 continue;
232 }
233 } else if constexpr (D == Direction::Backward) {
234 if (label->resources[TIME_INDEX] <= q_star[TIME_INDEX]) {
235 label->set_extended(true);
236 continue;
237 }
238 }
239 }
240
241 domin_smaller = false;
242
243 // Clear the visited buckets vector for the current label
244 std::memset(Bvisited.data(), 0, Bvisited.size() * sizeof(uint64_t));
245
246 // Check if the label is dominated by any labels in smaller buckets
247 domin_smaller =
248 DominatedInCompWiseSmallerBuckets<D, S>(label, bucket, c_bar, Bvisited, ordered_sccs);
249
250 if (!domin_smaller) {
251 // Lambda function to process new labels after extension
252 auto process_new_label = [&](Label *new_label) {
253 stat_n_labels++; // Increment number of labels processed
254
255 int &to_bucket = new_label->vertex; // Get the bucket to which the new label belongs
256 dominated = false;
257 const auto &to_bucket_labels =
258 buckets[to_bucket].get_labels(); // Get existing labels in the destination bucket
259
260#ifndef PSTEP
261 if constexpr (S == Stage::Four) {
262 // Track dominance checks for this bucket
263 if constexpr (D == Direction::Forward) {
264 dominance_checks_per_bucket[to_bucket] += to_bucket_labels.size();
265 }
266 }
267#endif
268 // Stage-specific dominance check
269 if constexpr (S == Stage::One) {
270 // If the new label has lower cost, remove dominated labels
271 for (auto *existing_label : to_bucket_labels) {
272 if (label->cost < existing_label->cost) {
273 buckets[to_bucket].remove_label(existing_label);
274 } else {
275 dominated = true;
276 break;
277 }
278 }
279 } else {
280#ifndef AVX
281
282 // General dominance check
283 for (size_t j = 0; j < to_bucket_labels.size(); ++j) {
284 Label *existing_label = to_bucket_labels[j];
285
286 // Prefetch the next label for comparison
287 //if (j + 1 < to_bucket_labels.size()) {
288 // __builtin_prefetch(to_bucket_labels[j + 1]);
289 //}
290
291 if (is_dominated<D, S>(new_label, existing_label)) {
292 stat_n_dom++; // Increment dominated labels count
293 dominated = true;
294 break;
295 }
296 }
297
298#else
299 if (check_dominance_against_vector<D, S>(new_label, to_bucket_labels, cut_storage)) {
300 stat_n_dom++; // Increment dominated labels count
301 dominated = true;
302 }
303#endif
304 }
305
306 if (!dominated) {
307 // Remove dominated labels from the bucket
308 if constexpr (S != Stage::Enumerate) {
309 std::vector<Label *> labels_to_remove;
310 for (auto *existing_label : to_bucket_labels) {
311 if (is_dominated<D, S>(existing_label, new_label)) {
312 labels_to_remove.push_back(existing_label);
313 }
314 }
315 // Now remove all marked labels in one pass
316 for (auto *label : labels_to_remove) { buckets[to_bucket].remove_label(label); }
317 }
318
319 n_labels++; // Increment the count of labels added
320
321 // Add the new label to the bucket
322#ifdef SORTED_LABELS
323 buckets[to_bucket].add_sorted_label(new_label);
324#elif LIMITED_BUCKETS
325 buckets[to_bucket].sorted_label(new_label, BUCKET_CAPACITY);
326#else
327 buckets[to_bucket].add_label(new_label);
328#endif
329 all_ext = false; // Not all labels have been extended, continue processing
330 }
331 };
332
333 a_ctr = 0;
334 // Process regular arcs for label extension
335 const auto &arcs = nodes[label->node_id].get_arcs<D>(scc_index);
336 for (const auto &arc : arcs) {
337 if constexpr (S == Stage::Four && F == Full::Partial) {
338 a_ctr++;
339 if (a_ctr > A_MAX) { break; }
340 }
341 Label *new_label = Extend<D, S, ArcType::Node, Mutability::Mut, F>(label, arc);
342 if (!new_label) {
343#ifdef UNREACHABLE_DOMINANCE
344 set_node_unreachable(label->unreachable_bitmap, arc.to);
345#endif
346 } else {
347 process_new_label(new_label); // Process the new label
348 }
349 }
350
351#ifdef FIX_BUCKETS
352 // Process jump arcs if in Stage 4
353 if constexpr (S == Stage::Four) {
354 const auto &jump_arcs = buckets[bucket].template get_jump_arcs<D>();
355 for (const auto &jump_arc : jump_arcs) {
356 Label *new_label = Extend<D, S, ArcType::Jump, Mutability::Const, F>(label, jump_arc);
357 if (!new_label) { continue; } // Skip if label extension failed
358 process_new_label(new_label); // Process the new label
359 }
360 }
361#endif
362 }
363
364 label->set_extended(true); // Mark the label as extended
365 }
366 }
367 } while (!all_ext); // Continue until all labels have been extended
368
369 // Update the cost bounds (c_bar) for the current SCC's buckets
370 for (const int bucket : sorted_sccs[scc_index]) {
371 const auto &labels = buckets[bucket].get_labels();
372
373 // Find the label with the minimum cost in the bucket
374 if (!labels.empty()) {
375 auto min_label = std::min_element(labels.begin(), labels.end(),
376 [](const Label *a, const Label *b) { return a->cost < b->cost; });
377 double min_cost = (*min_label)->cost;
378 c_bar[bucket] = std::min(c_bar[bucket], min_cost); // Update the lower bound for the bucket
379 }
380
381 // Update the cost bound for the bucket's dependencies (Phi buckets)
382 for (auto phi_bucket : Phi[bucket]) { c_bar[bucket] = std::min(c_bar[bucket], c_bar[phi_bucket]); }
383 }
384 }
385
386 // Get the best label from the topological order
387 Label *best_label = get_best_label<D>(topological_order, c_bar, sccs);
388
389 // Store the best label for forward or backward direction
390 if constexpr (D == Direction::Forward) {
391 fw_best_label = best_label;
392 } else {
393 bw_best_label = best_label;
394 }
395
396 return c_bar; // Return the cost bounds for each bucket
397}

403template <Stage S>
404std::vector<Label *> BucketGraph::bi_labeling_algorithm() {
405
406 // If in Stage 3, apply heuristic fixing based on q_star
407 if constexpr (S == Stage::Three) {
408 heuristic_fixing<S>();
409 }
410 // If in Stage 4, reset fixed buckets if it's the first reset
411 else if constexpr (S == Stage::Four) {
412 if (first_reset) {
413 reset_fixed();
414 first_reset = false; // Ensure this is only done once
415 }
416 }
417
418#ifdef FIX_BUCKETS
419 // If in Stage 4, apply bucket fixing based on q_star
420 if constexpr (S == Stage::Four) { bucket_fixing<S>(); }
421#endif
422
423 // Reset the label pool to ensure no leftover labels from previous runs
424 reset_pool();
425 // Perform any common initializations (data structures, etc.)
426 common_initialization();
427
428 // Initialize the cost bound vectors for forward and backward buckets
429 std::vector<double> forward_cbar(fw_buckets.size());
430 std::vector<double> backward_cbar(bw_buckets.size());
431
432 // Run the labeling algorithm in both directions, but only partially (Full::Partial)
433 run_labeling_algorithms<S, Full::Partial>(forward_cbar, backward_cbar);
434
435 // Acquire the best label from the forward label pool (will later combine with backward)
436 auto best_label = label_pool_fw.acquire();
437
438 // Check if the best forward and backward labels can be combined into a feasible solution
439 if (check_feasibility(fw_best_label, bw_best_label)) {
440 // If feasible, compute and combine the best forward and backward labels into one
441 best_label = compute_label(fw_best_label, bw_best_label);
442 } else {
443 // If not feasible, set the best label to have infinite cost (not usable)
444 best_label->cost = 0.0;
445 best_label->real_cost = std::numeric_limits<double>::infinity();
446 best_label->nodes_covered = {};
447 }
448
449 // Add the best label (combined forward/backward path) to the merged label list
450 merged_labels.push_back(best_label);
451
452 // For Stage Enumerate, print a message when labels are concatenated
453 if constexpr (S == Stage::Enumerate) { fmt::print("Labels generated, concatenating...\n"); }
454
455 // Setup visited buckets tracking for forward buckets
456 const size_t n_segments = fw_buckets_size / 64 + 1;
457 std::vector<uint64_t> Bvisited(n_segments, 0);
458
459 // Iterate over all forward buckets
460 for (auto bucket = 0; bucket < fw_buckets_size; ++bucket) {
461 auto &current_bucket = fw_buckets[bucket]; // Get the current bucket
462 const auto &labels = current_bucket.get_labels(); // Get labels in the current bucket
463 //
464 if constexpr (S == Stage::Four) {
465 non_dominated_labels_per_bucket += labels.size(); // Track non-dominated labels
466 }
467 // Process each label in the bucket
468 for (const Label *L : labels) {
469 // if (L->resources[TIME_INDEX] > q_star[TIME_INDEX]) { continue; } // Skip if label exceeds q_star
470
471 // Get arcs corresponding to nodes for this label (Forward direction)
472 const auto &to_arcs = nodes[L->node_id].get_arcs<Direction::Forward>();
473 // Iterate over each arc from the current node
474 for (const auto &arc : to_arcs) {
475 const auto &to_node = arc.to;
476
477 // Skip fixed arcs in Stage 3 if necessary
478 if constexpr (S == Stage::Three) {
479 if (fixed_arcs[L->node_id][to_node] == 1) {
480 continue; // Skip if the arc is fixed
481 }
482 }
483
484 // Attempt to extend the current label using this arc
485 auto L_prime = Extend<Direction::Forward, S, ArcType::Node, Mutability::Const, Full::Reverse>(L, arc);
486
487 // Note: apparently without the second condition it work better in some cases
488 // Check if the new label is valid and respects the q_star constraints
489 if (!L_prime || L_prime->resources[TIME_INDEX] <= q_star[TIME_INDEX]) {
490 continue; // Skip invalid labels or those that exceed q_star
491 }
492
493 // Get the bucket for the extended label
494 auto b_prime = L_prime->vertex;
495
496 // Clear visited buckets tracking for this new label extension
497 std::memset(Bvisited.data(), 0, Bvisited.size() * sizeof(uint64_t));
498
499 // Concatenate this new label with the best label found so far
500 ConcatenateLabel<S>(L, b_prime, best_label, Bvisited);
501 }
502 }
503 }
504
505 // Sort the merged labels by cost, to prioritize cheaper labels
506 pdqsort(merged_labels.begin(), merged_labels.end(),
507 [](const Label *a, const Label *b) { return a->cost < b->cost; });
508
509 inner_obj = merged_labels[0]->cost;
510
511#ifdef SCHRODINGER
512 // if merged_labels is bigger than 10, create Path related to the remaining ones
513 // and add them to a std::vector<Path>
514 if (merged_labels.size() > N_ADD) {
515 std::vector<Path> paths;
516 int labels_size = merged_labels.size();
517 for (size_t i = N_ADD; i < std::min(N_ADD + N_ADD, labels_size); ++i) {
518 if (merged_labels[i]->nodes_covered.size() <= 3) { continue; }
519 Path path = Path(merged_labels[i]->nodes_covered, merged_labels[i]->real_cost);
520 paths.push_back(path);
521 }
522 sPool.add_paths(paths);
523 sPool.iterate();
524 }
525#endif
526
527 // Return the final list of merged labels after processing
528
529#ifdef RIH
530 std::vector<Label *> top_labels;
531 top_labels.reserve(5); // Reserve memory for 5 elements
532 for (size_t i = 0; i < std::min(5, static_cast<int>(merged_labels.size())); ++i) {
533 top_labels.push_back(merged_labels[i]);
534 }
535
536 auto new_labels = ils->perturbation(top_labels, nodes);
537
538 // Use move semantics to avoid unnecessary copying of new_labels
539 merged_labels.insert(merged_labels.end(), std::make_move_iterator(new_labels.begin()),
540 std::make_move_iterator(new_labels.end()));
541
542 // Sort with inplace_merge to minimize sorting overhead
543 auto mid = merged_labels.end() - new_labels.size();
544 std::inplace_merge(merged_labels.begin(), mid, merged_labels.end(),
545 [](const Label *a, const Label *b) { return a->cost < b->cost; });
546
547#endif
548 return merged_labels;
549}

555template <Direction D, Stage S, ArcType A, Mutability M, Full F>
556inline Label *
557BucketGraph::Extend(const std::conditional_t<M == Mutability::Mut, Label *, const Label *> L_prime,
558 const std::conditional_t<A == ArcType::Bucket, BucketArc,
559 std::conditional_t<A == ArcType::Jump, JumpArc, Arc>> &gamma) noexcept {
560 // Get the forward or backward bucket structures, depending on the direction (D)
561 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
562 auto &label_pool = assign_buckets<D>(label_pool_fw, label_pool_bw);
563 auto &fixed_buckets = assign_buckets<D>(fw_fixed_buckets, bw_fixed_buckets);
564
565 // Precompute some values from the current label (L_prime) to avoid recalculating inside the loop
566 const int initial_node_id = L_prime->node_id;
567 auto initial_resources = L_prime->resources; // Copy the current label's resources
568 const double initial_cost = L_prime->cost; // Store the initial cost of the label
569
570 int node_id = -1; // Initialize node ID
571 // Determine the target node based on the arc type (Bucket, Node, or Jump)
572 if constexpr (A == ArcType::Bucket) {
573 node_id = buckets[gamma.to_bucket].node_id; // Use the node ID from the bucket
574 } else if constexpr (A == ArcType::Node) {
575 node_id = gamma.to; // Use the node ID from the arc
576 } else if constexpr (A == ArcType::Jump) {
577 node_id = buckets[gamma.jump_bucket].node_id; // Use the node ID from the jump bucket
578 // Update resources based on jump bucket bounds
579 for (size_t i = 0; i < initial_resources.size(); ++i) {
580 if constexpr (D == Direction::Forward) {
581 initial_resources[i] =
582 std::max(initial_resources[i], static_cast<double>(buckets[gamma.jump_bucket].lb[i]));
583 } else {
584 initial_resources[i] =
585 std::min(initial_resources[i], static_cast<double>(buckets[gamma.jump_bucket].ub[i]));
586 }
587 }
588 }
589
590 // Check if the arc between initial_node_id and node_id is fixed, and skip if so (in Stage 3)
591 if constexpr (S == Stage::Three) {
592 if constexpr (D == Direction::Forward) {
593 if (fixed_arcs[initial_node_id][node_id] == 1) { return nullptr; }
594 } else {
595 if (fixed_arcs[node_id][initial_node_id] == 1) { return nullptr; }
596 }
597 }
598
599 // Check if the node has already been visited for enumeration (Stage Enumerate)
600 if constexpr (S == Stage::Enumerate) {
601 if (is_node_visited(L_prime->visited_bitmap, node_id)) { return nullptr; }
602 }
603
604 // Perform 2-cycle elimination: if the node ID is the same as the current label's node, skip
605 if (node_id == L_prime->node_id) { return nullptr; }
606
607 // Check if node_id is in the neighborhood of initial_node_id and has already been visited
608 size_t segment = node_id >> 6; // Determine the segment in the bitmap
609 size_t bit_position = node_id & 63; // Determine the bit position in the segment
610 if constexpr (S != Stage::Enumerate) {
611 if ((neighborhoods_bitmap[initial_node_id][segment] & (1ULL << bit_position)) &&
612 is_node_visited(L_prime->visited_bitmap, node_id)) {
613 return nullptr; // Skip if the node is in the neighborhood and has been visited
614 }
615 }
616
617 // Get the VRP node corresponding to node_id
618 const VRPNode &VRPNode = nodes[node_id];
619
620 // Initialize new resources based on the arc's resource increments and check feasibility
621 std::vector<double> new_resources(initial_resources.size());
622
623 // Note: workaround
624 constexpr size_t N = R_SIZE;
625 if (!process_all_resources<D, 0, N>(new_resources, initial_resources, gamma, VRPNode)) {
626 return nullptr; // Handle failure case (constraint violation)
627 }
628
629#ifdef PSTEP
630 // counter the number of bits set in L_prime->visited_bitmap
631 size_t n_visited = 0;
632 for (size_t i = 0; i < L_prime->visited_bitmap.size(); ++i) {
633 n_visited += __builtin_popcountll(L_prime->visited_bitmap[i]);
634 }
635 if (n_visited >= options.max_path_size) { return nullptr; }
636#endif
637
638 // Get the bucket number for the new node and resource state
639 int to_bucket = get_bucket_number<D>(node_id, new_resources);
640
641#ifdef FIX_BUCKETS
642 // Skip if the bucket is fixed (in Stage 4) and not a jump arc
643 if constexpr (S == Stage::Four && A != ArcType::Jump) {
644 if (fixed_buckets[L_prime->vertex][to_bucket] == 1) { return nullptr; }
645 }
646#endif
647
648 // Compute travel cost between the initial and current nodes
649 const double travel_cost = getcij(initial_node_id, node_id);
650#ifndef PSTEP
651 double new_cost = initial_cost + travel_cost - VRPNode.cost;
652#else
653 double new_cost = initial_cost + travel_cost;
654#endif
655
656 // Compute branching duals
657 if (branching_duals->size() > 0) {
658 if constexpr (D == Direction::Forward) {
659 new_cost -= branching_duals->getDual(initial_node_id, node_id);
660 } else {
661 new_cost -= branching_duals->getDual(node_id, initial_node_id);
662 }
663 new_cost -= branching_duals->getDual(node_id);
664 }
665
666#if defined(RCC) || defined(EXACT_RCC)
667 if constexpr (S == Stage::Four) {
668 if constexpr (D == Direction::Forward) {
669 auto arc_dual = arc_duals.getDual(initial_node_id, node_id);
670 new_cost -= arc_dual;
671 } else if constexpr (D == Direction::Backward) {
672 auto arc_dual = arc_duals.getDual(node_id, initial_node_id);
673 new_cost -= arc_dual;
674 }
675 }
676#endif
677
678#ifdef KP_BOUND
679 // Apply knapsack bound check in the forward direction (if applicable)
680 if constexpr (D == Direction::Forward) {
681 auto kpBound = knapsackBound(L_prime);
682 if (kpBound > 0.0) {
683 return nullptr; // Skip if knapsack bound is exceeded
684 }
685 }
686#endif
687
688 // Acquire a new label from the pool and initialize it with the new state
689 auto new_label = label_pool.acquire();
690 new_label->initialize(to_bucket, new_cost, new_resources, node_id);
691
692 if constexpr (F == Full::Reverse) {
693 return new_label; // Return the new label early if in reverse mode
694 }
695
696 // new_label->visited_bitmap = L_prime->visited_bitmap; // Copy visited bitmap from the original label
697 std::memcpy(new_label->visited_bitmap.data(), L_prime->visited_bitmap.data(),
698 new_label->visited_bitmap.size() * sizeof(uint64_t));
699 set_node_visited(new_label->visited_bitmap, node_id); // Mark the new node as visited
700
701#ifdef UNREACHABLE_DOMINANCE
702 // Copy unreachable bitmap (if applicable)
703 new_label->unreachable_bitmap = L_prime->unreachable_bitmap;
704#endif
705
706 // Update real cost (for tracking the total travel cost)
707 new_label->real_cost = L_prime->real_cost + travel_cost;
708
709 // Set the parent label, depending on mutability
710 if constexpr (M == Mutability::Mut) { new_label->parent = L_prime; }
711
712#if defined(SRC3) || defined(SRC)
713 // Copy the SRC map from the original label (for SRC cuts)
714 new_label->SRCmap = L_prime->SRCmap;
715#endif
716
717 // If not in enumeration stage, update visited bitmap to avoid redundant labels
718 if constexpr (S != Stage::Enumerate) {
719 size_t limit = new_label->visited_bitmap.size();
720 for (size_t i = 0; i < limit; ++i) {
721 uint64_t current_visited = new_label->visited_bitmap[i];
722
723 if (!current_visited) continue; // Skip if no nodes were visited in this segment
724
725 uint64_t neighborhood_mask = neighborhoods_bitmap[node_id][i]; // Get neighborhood mask for the current node
726 uint64_t bits_to_clear = current_visited & ~neighborhood_mask; // Determine which bits to clear
727
728 if (i == node_id / 64) {
729 bits_to_clear &= ~(1ULL << (node_id & 63)); // Ensure current node remains visited
730 }
731
732 new_label->visited_bitmap[i] &= ~bits_to_clear; // Clear irrelevant visited nodes
733 }
734 }
735
736#if defined(SRC3) || defined(SRC)
737 // Apply SRC (Subset Row Cuts) logic in Stages 3, 4, and Enumerate
738 if constexpr (S == Stage::Four || S == Stage::Enumerate) {
739 auto &cutter = cut_storage; // Access the cut storage manager
740 auto &SRCDuals = cutter->SRCDuals; // Access the dual values for the SRC cuts
741 const uint64_t bit_mask = 1ULL << bit_position; // Precompute bit shift for the node's position
742
743 for (std::size_t idx = 0; idx < cutter->size(); ++idx) {
744 auto it = cutter->begin();
745 std::advance(it, idx);
746 const auto &cut = *it;
747 const auto &baseSet = cut.baseSet;
748 const auto &baseSetorder = cut.baseSetOrder;
749 const auto &neighbors = cut.neighbors;
750 const auto &multipliers = cut.p;
751
752#if defined(SRC3)
753 // Apply SRC3 logic: if the node is in the base set, increment the SRC map
754 bool bitIsSet3 = baseSet[segment] & bit_mask;
755 if (bitIsSet3) {
756 new_label->SRCmap[idx]++;
757 if (new_label->SRCmap[idx] % 2 == 0) { new_label->cost -= SRCDuals[idx]; }
758 }
759#endif
760
761#if defined(SRC)
762 // Apply SRC logic: Update the SRC map based on neighbors and base set
763 bool bitIsSet = neighbors[segment] & bit_mask;
764 bool bitIsSet2 = baseSet[segment] & bit_mask;
765
766 auto &src_map_value = new_label->SRCmap[idx]; // Use reference to avoid multiple accesses
767 if (!bitIsSet) {
768 src_map_value = 0.0; // Reset the SRC map value
769 }
770
771 if (bitIsSet2) {
772 src_map_value += multipliers.num[baseSetorder[node_id]];
773 if (src_map_value % multipliers.den == 0) {
774 new_label->cost -= SRCDuals[idx]; // Apply the SRC dual value if threshold is exceeded
775 }
776 }
777#endif
778 }
779 }
780#endif
781
782 // Return the newly created and initialized label
783 return new_label;
784}

790template <Direction D, Stage S>
791inline bool BucketGraph::is_dominated(const Label *new_label, const Label *label) noexcept {
792
793 // Extract resources for the new label and the comparison label
794 const auto &new_resources = new_label->resources;
795 const auto &label_resources = label->resources;
796
797 double sumSRC = 0;
798
799#ifdef SRC
800 if constexpr (S == Stage::Four || S == Stage::Enumerate) {
801 const auto &SRCDuals = cut_storage->SRCDuals;
802 if (!SRCDuals.empty()) {
803 const auto &labelSRCMap = label->SRCmap;
804 const auto &newLabelSRCMap = new_label->SRCmap;
805 for (size_t i = 0; i < SRCDuals.size(); ++i) {
806 const auto &den = cut_storage->getCut(i).p.den;
807 const auto labelMod = labelSRCMap[i] % den;
808 const auto newLabelMod = newLabelSRCMap[i] % den;
809 if (labelMod > newLabelMod) { sumSRC += SRCDuals[i]; }
810 }
811 }
812 if (label->cost - sumSRC > new_label->cost) { return false; }
813 } else
814#endif
815 {
816 // Simple cost check: if the comparison label has a higher cost, it is not dominated
817 if (label->cost > new_label->cost) { return false; }
818 }
819
820 // Check resource conditions based on the direction (Forward or Backward)
821 for (size_t i = 0; i < R_SIZE; ++i) {
822 if constexpr (D == Direction::Forward) {
823 // In Forward direction: the comparison label must not have more resources than the new label
824 if (label_resources[i] > new_resources[i]) { return false; }
825 } else if constexpr (D == Direction::Backward) {
826 // In Backward direction: the comparison label must not have fewer resources than the new label
827 if (label_resources[i] < new_resources[i]) { return false; }
828 }
829 }
830
831 // TODO:: check again this unreachable dominance
832#ifndef UNREACHABLE_DOMINANCE
833 // Check visited nodes (bitmap comparison) for Stages 3, 4, and Enumerate
834 if constexpr (S == Stage::Three || S == Stage::Four || S == Stage::Enumerate) {
835 // Iterate through the visited bitmap and ensure that the new label visits all nodes that the comparison
836 // label visits
837 for (size_t i = 0; i < label->visited_bitmap.size(); ++i) {
838 // If the comparison label visits a node that the new label does not, it is not dominated
839 if ((label->visited_bitmap[i] & ~new_label->visited_bitmap[i]) != 0) { return false; }
840 }
841 }
842#else
843 // Unreachable dominance logic: check visited and unreachable nodes in Stages 3, 4, and Enumerate
844 if constexpr (S == Stage::Three || S == Stage::Four || S == Stage::Enumerate) {
845 for (size_t i = 0; i < label->visited_bitmap.size(); ++i) {
846 // Combine visited and unreachable nodes in the comparison label's bitmap
847 auto combined_label_bitmap = label->visited_bitmap[i] | label->unreachable_bitmap[i];
848 // Ensure the new label visits all nodes that the comparison label (or its unreachable nodes) visits
849 if ((combined_label_bitmap & ~new_label->visited_bitmap[i]) != 0) { return false; }
850 }
851 }
852#endif
853
854#ifdef SRC3
855 // Additional SRC3 logic for Stages 3, 4, and Enumerate
856 if constexpr (S == Stage::Four || S == Stage::Enumerate) {
857 const auto &SRCDuals = cut_storage->SRCDuals;
858 if (!SRCDuals.empty()) {
859 sumSRC = 0; // Reset sumSRC for SRC3 logic
860 // Iterate through SRC duals
861 for (size_t i = 0; i < SRCDuals.size(); ++i) {
862 // Check if the comparison label's SRC map is greater in modulo 2 sense
863 if ((label->SRCmap[i] % 2) > (new_label->SRCmap[i] % 2)) { sumSRC += SRCDuals[i]; }
864 }
865 // If the adjusted cost of the comparison label is greater, it is not dominated
866 if (label->cost + sumSRC > new_label->cost) { return false; }
867 }
868 }
869#endif
870
871 // If all conditions are met, return true, indicating that the new label is dominated by the comparison label
872 return true;
873}

898template <Direction D, Stage S>
899inline bool BucketGraph::DominatedInCompWiseSmallerBuckets(const Label *L, int bucket, const std::vector<double> &c_bar,
900 std::vector<uint64_t> &Bvisited,
901 const std::vector<std::vector<int>> &bucket_order) noexcept {
902 // Assign the appropriate buckets and Phi structures based on direction (D)
903 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
904 auto &Phi = assign_buckets<D>(Phi_fw, Phi_bw);
905
906 const int b_L = L->vertex; // The vertex (bucket) associated with the label L
907 std::vector<int> bucketStack; // Stack to manage the traversal of buckets
908 bucketStack.reserve(10);
909 bucketStack.push_back(bucket); // Start with the input bucket
910
911 auto &uf = assign_buckets<D>(fw_union_find, bw_union_find);
912 // Traverse the graph of buckets in a depth-first manner
913 while (!bucketStack.empty()) {
914 int currentBucket = bucketStack.back(); // Get the bucket at the top of the stack
915 bucketStack.pop_back(); // Remove it from the stack
916
917 // Mark the current bucket as visited by updating the Bvisited bitmask
918 const size_t segment = currentBucket >> 6; // Determine the segment for the current bucket
919 const size_t bit_position = currentBucket & 63; // Determine the bit position within the segment
920 Bvisited[segment] |= (1ULL << bit_position); // Set the bit corresponding to the current bucket as visited
921
922 // Check if the label's cost is lower than the cost bound (c_bar) for the current bucket
923 // and if the current bucket precedes the label's bucket according to the bucket order
924 if (L->cost < c_bar[currentBucket] && ::precedes<int>(bucket_order, currentBucket, b_L, uf)) {
925 return false; // The label is not dominated, return false early
926 }
927
928 // If the current bucket is different from the label's bucket, compare the labels in this bucket
929 if (b_L != currentBucket) {
930 const auto &bucket_labels = buckets[currentBucket].get_labels(); // Get the labels in the current bucket
931 // Iterate over each label in the current bucket and check if it dominates L
932
933#ifndef AVX
934
935 for (auto *label : bucket_labels) {
936 if (is_dominated<D, S>(L, label)) {
937 return true; // If any label dominates L, return true
938 }
939 }
940#else
941 if (check_dominance_against_vector<D, S>(L, bucket_labels, cut_storage)) {
942 return true; // If any label dominates L, return true
943 }
944#endif
945 }
946
947 // Add the neighboring buckets (from Phi) to the stack if they haven't been visited yet
948 for (const int b_prime : Phi[currentBucket]) {
949 const size_t segment_prime = b_prime >> 6; // Determine the segment for the neighboring bucket
950 const size_t bit_position_prime = b_prime & 63; // Determine the bit position within the segment
951
952 // If the neighboring bucket hasn't been visited, push it onto the stack
953 if ((Bvisited[segment_prime] & (1ULL << bit_position_prime)) == 0) { bucketStack.push_back(b_prime); }
954 }
955 }
956
957 // If no domination was found in any smaller bucket, return false
958 return false;
959}

969template <Stage state, Full fullness>
970void BucketGraph::run_labeling_algorithms(std::vector<double> &forward_cbar, std::vector<double> &backward_cbar) {
971 // Create tasks for forward and backward labeling algorithms
972
973 auto forward_task = stdexec::schedule(bi_sched) |
974 stdexec::then([&]() { return labeling_algorithm<Direction::Forward, state, fullness>(); });
975
976 auto backward_task = stdexec::schedule(bi_sched) |
977 stdexec::then([&]() { return labeling_algorithm<Direction::Backward, state, fullness>(); });
978
979 // Execute the tasks in parallel and synchronize
980 auto work = stdexec::when_all(std::move(forward_task), std::move(backward_task)) |
981 stdexec::then([&](auto forward_result, auto backward_result) {
982 forward_cbar = std::move(forward_result);
983 backward_cbar = std::move(backward_result);
984 });
985
986 stdexec::sync_wait(std::move(work));
987}
