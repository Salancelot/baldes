138Label *BucketGraph::compute_label(const Label *L, const Label *L_prime) {
139 double cij_cost = getcij(L->node_id, L_prime->node_id);
140 double new_cost = L->cost + L_prime->cost + cij_cost;
141
142 double real_cost = L->real_cost + L_prime->real_cost + cij_cost;
143
144#if defined(RCC) || defined(EXACT_RCC)
145 auto arc_dual = arc_duals.getDual(L->node_id, L_prime->node_id);
146 new_cost -= arc_dual;
147#endif
148
149 // Branching duals
150 if (branching_duals->size() > 0) { new_cost -= branching_duals->getDual(L->node_id, L_prime->node_id); }
151
152 // Directly acquire new_label and set the cost
153 auto new_label = label_pool_fw.acquire();
154 new_label->cost = new_cost;
155 new_label->real_cost = real_cost;
156
157#ifdef SRC
158 // Check SRCDuals condition for specific stages
159 auto sumSRC = 0.0;
160 const auto &SRCDuals = cut_storage->SRCDuals;
161 if (!SRCDuals.empty()) {
162 size_t idx = 0;
163 auto sumSRC =
164 std::transform_reduce(SRCDuals.begin(), SRCDuals.end(), 0.0, std::plus<>(), [&](const auto &dual) {
165 size_t curr_idx = idx++;
166 auto den = cut_storage->getCut(curr_idx).p.den;
167 auto sum = (L->SRCmap[curr_idx] + L_prime->SRCmap[curr_idx]);
168 return (sum >= den) ? dual : 0.0;
169 });
170
171 new_label->cost -= sumSRC;
172 }
173
174#endif
175#ifdef SRC3
176 // Check SRCDuals condition for specific stages
177 auto sumSRC = 0.0;
178 const auto &SRCDuals = cut_storage->SRCDuals;
179 if (!SRCDuals.empty()) {
180 double sumSRC = 0;
181 for (size_t i = 0; i < SRCDuals.size(); ++i) {
182 if (SRCDuals[i] != 0 && (L->SRCmap[i] % 2 + L_prime->SRCmap[i] % 2 >= 1)) { sumSRC += SRCDuals[i]; }
183 }
184 }
185 new_label->cost -= sumSRC;
186#endif
187
188 new_label->nodes_covered.clear();
189
190 // Start by inserting backward list elements
191 size_t forward_size = 0;
192 auto L_bw = L_prime;
193 for (; L_bw != nullptr; L_bw = L_bw->parent) {
194 new_label->nodes_covered.push_back(L_bw->node_id); // Insert backward elements directly
195 }
196
197 // Now insert forward list elements in reverse order without using std::reverse
198 auto L_fw = L;
199 for (; L_fw != nullptr; L_fw = L_fw->parent) {
200 new_label->nodes_covered.insert(new_label->nodes_covered.begin(),
201 L_fw->node_id); // Insert forward elements at the front
202 forward_size++;
203 }
204
205 return new_label;
206}

212std::vector<int> BucketGraph::computePhi(int &bucket_id, bool fw) {
213 std::vector<int> phi;
214
215 // Ensure bucket_id is within valid bounds
216 auto &buckets = fw ? fw_buckets : bw_buckets;
217 auto &fixed_buckets = fw ? fw_fixed_buckets : bw_fixed_buckets;
218 auto &node_interval_trees = fw ? fw_node_interval_trees : bw_node_interval_trees;
219
220 if constexpr (R_SIZE > 1) {
221 if (bucket_id >= buckets.size() || bucket_id < 0) return phi;
222
223 std::vector<double> total_ranges(intervals.size());
224 std::vector<double> base_intervals(intervals.size());
225
226 for (int r = 0; r < intervals.size(); ++r) {
227 total_ranges[r] = R_max[r] - R_min[r]; // Ensure integer type for total range
228 base_intervals[r] = total_ranges[r] / intervals[r].interval;
229 }
230
231 // Get the node ID and current bucket
232 int node_id = buckets[bucket_id].node_id;
233 auto &current_bucket = buckets[bucket_id];
234
235 // Retrieve the pre-built Splay Tree for this node
236 auto &node_tree = node_interval_trees[node_id];
237
238 // Search for matching intervals using the existing Splay Tree
239 if (fw) {
240 // Forward search: find the interval just below the current bucket
241 std::vector<double> target_low = current_bucket.lb;
242 for (int r = 0; r < intervals.size(); ++r) {
243 target_low[r] -= base_intervals[r]; // Adjust for the base intervals
244 }
245
246 TreeNode *found_node = node_tree.find(target_low);
247 if (found_node != nullptr && buckets[found_node->bucket_index].node_id == node_id) {
248 // Check if the found bucket is fixed
249#ifdef FIX_BUCKETS
250 if (fixed_buckets[found_node->bucket_index][bucket_id] == 0)
251#endif
252 {
253 phi.push_back(found_node->bucket_index);
254 }
255 }
256 } else {
257 // Backward search: find the interval just above the current bucket
258 std::vector<double> target_high = current_bucket.ub;
259 for (int r = 0; r < intervals.size(); ++r) {
260 target_high[r] += base_intervals[r]; // Adjust for the base intervals
261 }
262
263 TreeNode *found_node = node_tree.find(target_high);
264 if (found_node != nullptr && buckets[found_node->bucket_index].node_id == node_id) {
265 // Check if the found bucket is fixed
266#ifdef FIX_BUCKETS
267 if (fixed_buckets[found_node->bucket_index][bucket_id] == 0)
268#endif
269 {
270 phi.push_back(found_node->bucket_index);
271 }
272 }
273 }
274
275 } else {
276 // Handle the case where R_SIZE == 1 with a simpler approach
277 int smaller = bucket_id - 1;
278
279 if (smaller >= 0 && buckets[smaller].node_id == buckets[bucket_id].node_id) {
280#ifdef FIX_BUCKETS
281 if (fixed_buckets[smaller][bucket_id] == 0)
282#endif
283 {
284 phi.push_back(smaller);
285 }
286 }
287 }
288 return phi;
289}

295void BucketGraph::calculate_neighborhoods(size_t num_closest) {
296 size_t num_nodes = nodes.size();
297
298 // Initialize the neighborhood bitmaps as vectors of uint64_t for forward and backward neighborhoods
299 neighborhoods_bitmap.resize(num_nodes); // Forward neighborhood
300
301 for (size_t i = 0; i < num_nodes; ++i) {
302 std::vector<std::pair<double, size_t>> forward_distances; // Distances for forward neighbors
303
304 for (size_t j = 0; j < num_nodes; ++j) {
305 if (i != j) {
306 // Forward distance (i -> j)
307 double forward_distance = getcij(i, j);
308 forward_distances.push_back({forward_distance, j});
309 }
310 }
311
312 // Sort distances to find the closest nodes
313 pdqsort(forward_distances.begin(), forward_distances.end());
314
315 // Initialize the neighborhood bitmap vector for node i (forward and backward)
316 size_t num_segments = (num_nodes + 63) / 64;
317 neighborhoods_bitmap[i].resize(num_segments, 0); // Resizing for forward bitmap
318
319 // Include the node itself in both forward and backward neighborhoods
320 size_t segment_self = i >> 6;
321 size_t bit_position_self = i & 63;
322 neighborhoods_bitmap[i][segment_self] |= (1ULL << bit_position_self); // Forward
323
324 // Map the top 'num_closest' closest nodes for forward and set them in the backward neighborhoods
325 for (size_t k = 0; k < num_closest && k < forward_distances.size(); ++k) {
326 size_t node_index = forward_distances[k].second;
327
328 // Determine the segment and the bit within the segment for the node_index (forward)
329 size_t segment = node_index >> 6;
330 size_t bit_position = node_index & 63;
331 neighborhoods_bitmap[i][segment] |= (1ULL << bit_position); // Forward neighbor
332 }
333 }

345void BucketGraph::augment_ng_memories(std::vector<double> &solution, std::vector<Path> &paths, bool aggressive,
346 int eta1, int eta2, int eta_max, int nC) {
347 std::set<std::pair<int, int>> forbidden_augmentations;
348 std::vector<std::vector<int>> cycles;
349
350 for (int col = 0; col < paths.size(); ++col) {
351
352 if (solution[col] > 1e-2 && solution[col] < 1 - 1e-2) {
353 ankerl::unordered_dense::map<int, int> visited_clients;
354 std::vector<int> cycle;
355 bool has_cycle = false;
356
357 for (int i = 0; i < paths[col].size(); ++i) {
358 int client = paths[col][i];
359 if (client == 0 || client == N_SIZE - 1) {
360 continue; // Ignore 0 in cycle detection
361 }
362 if (visited_clients.find(client) != visited_clients.end()) {
363 has_cycle = true;
364 // Start from the first occurrence of the repeated client to form the cycle
365 for (int j = visited_clients[client]; j <= i; ++j) { cycle.push_back(paths[col][j]); }
366 break; // Stop once the cycle is stored
367 }
368 visited_clients[client] = i;
369 }
370
371 if (has_cycle) { cycles.push_back(cycle); }
372 }
373 }
374
375 // Sort cycles by size to prioritize smaller cycles
376 pdqsort(cycles.begin(), cycles.end(),
377 [](const std::vector<int> &a, const std::vector<int> &b) { return a.size() < b.size(); });
378 int forbidden_count = 0;
379
380 for (const auto &cycle : cycles) {
381 // Check the current sizes of neighborhoods involved in the cycle
382 bool can_forbid = true;
383 for (const auto &node : cycle) {
384 // Count the number of 1s in neighborhoods_bitmap[node]
385 int count = 0;
386 // print size of neighborhoods_bitmap
387 for (const auto &segment : neighborhoods_bitmap[node]) {
388 count += __builtin_popcountll(segment); // Counts the number of set bits (1s)
389 if (count >= eta_max) {
390 can_forbid = false;
391 break;
392 }
393 }
394 if (!can_forbid) { break; }
395 }
396
397 if (can_forbid && (cycle.size() <= eta1 || (forbidden_count < eta2 && !cycle.empty()))) {
398 // Forbid the cycle
399 forbidCycle(cycle, aggressive);
400 forbidden_count++;
401 }
402
403 if (forbidden_count >= eta2) { break; }
404 }

415void BucketGraph::forbidCycle(const std::vector<int> &cycle, bool aggressive) {
416 for (size_t i = 0; i < cycle.size() - 1; ++i) {
417 int v1 = cycle[i];
418 int v2 = cycle[i + 1];
419
420 // Update the bitmap to forbid v2 in the neighborhood of v1
421 size_t segment = v2 >> 6;
422 size_t bit_position = v2 & 63;
423 neighborhoods_bitmap[v1][segment] |= (1ULL << bit_position);
424
425 if (aggressive) {
426 segment = v1 >> 6;
427 bit_position = v1 & 63;
428 neighborhoods_bitmap[v2][segment] |= (1ULL << bit_position);
429 }
430 }

711void BucketGraph::mono_initialization() {
712 // Clear previous data
713 merged_labels.clear();
714 merged_labels.reserve(100);
715 fw_c_bar.clear();
716
717 // Resize cost vectors to match the number of buckets
718 fw_c_bar.resize(fw_buckets.size(), std::numeric_limits<double>::infinity());
719
720 auto &num_buckets = assign_buckets<Direction::Forward>(num_buckets_fw, num_buckets_bw);
721 auto &num_bucket_index = assign_buckets<Direction::Forward>(num_buckets_index_fw, num_buckets_index_bw);
722
723 int num_intervals = intervals.size(); // Determine how many resources we have (number of intervals)
724 std::vector<double> total_ranges(num_intervals);
725 std::vector<double> base_intervals(num_intervals);
726
727 // Calculate base intervals and total ranges for each resource dimension
728 for (int r = 0; r < intervals.size(); ++r) {
729 total_ranges[r] = R_max[r] - R_min[r];
730 base_intervals[r] = total_ranges[r] / intervals[r].interval;
731 }
732
733 // Clear forward and backward buckets
734 for (auto b = 0; b < fw_buckets_size; b++) { fw_buckets[b].clear(); }
735
736 auto &VRPNode = nodes[0]; // Example for the first node
737
738 std::vector<int> node_total_ranges(num_intervals);
739 for (int r = 0; r < num_intervals; ++r) { node_total_ranges[r] = VRPNode.ub[r] - VRPNode.lb[r]; }
740
741 // Helper lambda to update current position of the intervals
742 auto update_position = [&](std::vector<int> &current_pos) -> bool {
743 bool done = true;
744 for (int r = num_intervals - 1; r >= 0; --r) {
745 current_pos[r]++;
746 if (current_pos[r] * base_intervals[r] < node_total_ranges[r]) {
747 done = false;
748 break;
749 } else {
750 current_pos[r] = 0;
751 }
752 }
753 return done;
754 };
755
756 auto calculate_index = [&](const std::vector<int> &current_pos, int &total_buckets) -> int {
757 int index = 0;
758
759 // Loop through each interval (dimension) and compute the index
760 for (int r = 0; r < current_pos.size(); ++r) {
761 index += current_pos[r]; // Accumulate the positional index across intervals
762 }
763
764 return index;
765 };
766
767 // Initialize forward buckets (generic for multiple dimensions)
768 std::vector<int> current_pos(num_intervals, 0);
769
770 // Iterate over all intervals for the forward direction
771 while (true) {
772 auto depot = label_pool_fw.acquire();
773 // print num_intervals
774 std::vector<double> interval_starts(num_intervals);
775 for (int r = 0; r < num_intervals; ++r) {
776 interval_starts[r] = std::min(R_max[r], VRPNode.lb[r] + current_pos[r] * base_intervals[r]);
777 }
778
779 // Adjust to calculate index using `num_buckets[0]`, which is likely multi-dimensional for the depot
780 int calculated_index = calculate_index(current_pos, num_buckets[options.depot]) +
781 num_bucket_index[options.depot]; // Calculate index once
782 depot->initialize(calculated_index, 0.0, interval_starts, options.depot);
783 depot->is_extended = false;
784 set_node_visited(depot->visited_bitmap, options.depot);
785#ifdef SRC
786 depot->SRCmap.assign(cut_storage->SRCDuals.size(), 0);
787#endif
788 fw_buckets[calculated_index].add_label(depot);
789 fw_buckets[calculated_index].node_id = options.depot;
790
791 if (update_position(current_pos)) break; // Update position and break if done
792 }
793}

808double BucketGraph::knapsackBound(const Label *l) {
809 Knapsack kp;
810 int rload = R_max[DEMAND_INDEX] - l->resources[DEMAND_INDEX];
811 kp.setCapacity(rload);
812
813 for (int i = 1; i < nodes.size(); ++i) {
814 if (!l->visits(i) && nodes[i].consumption[DEMAND_INDEX] <= rload) {
815 kp.addItem(nodes[i].cost, nodes[i].consumption[DEMAND_INDEX]);
816 }
817 }
818
819 return l->cost - kp.solve();
820}

1004 new_label->cost = L->cost; // Use the cost from L
1005 new_label->real_cost = L->real_cost; // Use the real cost from L
1006
1007 // Calculate the number of nodes covered by the label (its ancestors)
1008 size_t label_size = 0;
1009 for (auto current_label = L; current_label != nullptr; current_label = current_label->parent) { label_size++; }
1010
1011 // Reserve space in one go
1012 new_label->nodes_covered.clear();
1013 new_label->nodes_covered.reserve(label_size);
1014
1015 // Insert the nodes from the label and its ancestors
1016 for (auto current_label = L; current_label != nullptr; current_label = current_label->parent) {
1017 new_label->nodes_covered.push_back(current_label->node_id);
1018 }
1019
1020 std::reverse(new_label->nodes_covered.begin(), new_label->nodes_covered.end());
