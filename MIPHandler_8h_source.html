128 baldes::Constraint *add_constraint(baldes::Constraint *constraint, const std::string &name) {
129 int constraint_index = constraints.size(); // Get the current index
130 constraint->set_index(constraint_index);
131 constraint->set_name(name);
132
133 // Add the constraint to the list
134 constraints.push_back(constraint);
135 b_vec.push_back(constraint->get_rhs());
136 // Get the expression and relation (assuming it needs to be used later)
137 const LinearExpression &expression = constraint->get_expression();
138

140 int row_index = constraint_index;

143
144 int col_index = var_name_to_index[var_name];
145 sparse_matrix.insert(row_index, col_index, coeff);

202 if (constraintIndex < 0 || constraintIndex >= constraints.size()) {

206 // Get the constraint that is being modified

211 for (int i = 0; i < values.size(); ++i) {

219 // Update or remove terms in the LinearExpression

281 GRBModel gurobiModel(env);
282
283 // Map to store Gurobi variables

285
286 // check if we have repeted variable name

288 // Step 1: Add variables to the Gurobi model

290 // Add each variable to the Gurobi model, according to its type and bounds
291 GRBVar gurobiVar = gurobiModel.addVar(var->get_lb(), var->get_ub(), var->get_objective_coefficient(),
292 toGRBVarType(var->get_type()), var->get_name());
293 gurobiVars[var->get_name()] = gurobiVar;
294 }
295
296 // Step 2: Add constraints to the Gurobi model
297 for (const auto &constraint : constraints) {
298 GRBLinExpr gurobiExpr = convertToGurobiExpr(constraint, gurobiVars);
299 if (constraint->get_relation() == '<') {
300 gurobiModel.addConstr(gurobiExpr <= constraint->get_rhs(), constraint->get_name());
301 } else if (constraint->get_relation() == '>') {
302 gurobiModel.addConstr(gurobiExpr >= constraint->get_rhs(), constraint->get_name());
303 } else if (constraint->get_relation() == '=') {
304 gurobiModel.addConstr(gurobiExpr == constraint->get_rhs(), constraint->get_name());
305 }
306 }
307
308 // Step 3: Set objective if needed (assuming a linear objective function)
309 GRBLinExpr objective;
310 for (const auto &var : variables) {
311 objective += gurobiVars[var->get_name()] * var->get_objective_coefficient();
312 }
313 gurobiModel.setObjective(objective, GRB_MINIMIZE); // Assume minimization problem
314 gurobiModel.update();

348 // Create a new HiGHS model
349 Highs highs;
350 HighsModel highsModel;
351
352 int numVars = variables.size();
353 int numConstrs = constraints.size();
354
355 // Set up the number of variables and constraints
356 highsModel.lp_.num_col_ = numVars;
357 highsModel.lp_.num_row_ = numConstrs;
358
359 // Resize vectors for variable bounds and objective coefficients
360 highsModel.lp_.col_lower_.resize(numVars);
361 highsModel.lp_.col_upper_.resize(numVars);
362 highsModel.lp_.col_cost_.resize(numVars);
363
364 // Set variable bounds and cost
365 int var_index = 0;
366 for (const auto &var : variables) {
367 highsModel.lp_.col_lower_[var_index] = var->get_lb();
368 highsModel.lp_.col_upper_[var_index] = var->get_ub();
369 highsModel.lp_.col_cost_[var_index] = var->get_objective_coefficient();
370 var_index++;
371 }
372
373 // Resize vectors for constraint bounds
374 highsModel.lp_.row_lower_.resize(numConstrs);
375 highsModel.lp_.row_upper_.resize(numConstrs);
376
377 // Set constraint bounds
378 int row_index = 0;
379 for (const auto &constraint : constraints) {
380 double lower_bound, upper_bound;
381 char relation = constraint->get_relation();
382 double rhs = constraint->get_rhs();
383
384 // Set lower and upper bounds for each constraint based on its type
385 if (relation == '<') {
386 lower_bound = -kHighsInf;
387 upper_bound = rhs;
388 } else if (relation == '>') {
389 lower_bound = rhs;
390 upper_bound = kHighsInf;
391 } else if (relation == '=') {
392 lower_bound = upper_bound = rhs;
393 }
394
395 highsModel.lp_.row_lower_[row_index] = lower_bound;
396 highsModel.lp_.row_upper_[row_index] = upper_bound;
397 row_index++;
398 }
399
400 // Assign sparse matrix data directly from `sparse_matrix`
401 highsModel.lp_.a_matrix_.start_ = sparse_matrix.getRowStart();
402 highsModel.lp_.a_matrix_.index_ = sparse_matrix.getIndices();
403 highsModel.lp_.a_matrix_.value_ = sparse_matrix.getValues();
404 highsModel.lp_.a_matrix_.format_ = MatrixFormat::kRowwise;
405
406 // Set the objective direction (assuming minimization)
407 highsModel.lp_.sense_ = ObjSense::kMinimize;
408
409 // Add the model to HiGHS
410 highs.passModel(highsModel);
411
412 return highsModel;
413 }

499 double getSlack(int row, const std::vector<double> &solution) {
500 // Ensure the sparse matrix row structure is built (we stay in COO mode)
501 auto rhs = constraints[row]->get_rhs(); // Get the right-hand side value for this row
502
503 // Compute the dot product of the solution vector and the specified row
504 double row_value = 0.0;
505
506 // Iterate through the non-zero elements in COO format
507 for (size_t i = 0; i < sparse_matrix.rows.size(); ++i) {
508 if (sparse_matrix.rows[i] == row) {
509 int col_index = sparse_matrix.cols[i]; // Get the column index for the current element
510 double value = sparse_matrix.values[i]; // Get the value of the current element
511
512 // Perform the dot product with the corresponding element in the solution vector
513 row_value += value * solution[col_index];
514 }
515 }
516
517 // Return the slack: rhs - (dot product of the row and solution)
518 return rhs - row_value;
519 }
