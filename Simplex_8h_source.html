22 // Perform LU decomposition with partial pivoting
23 void decompose(const Eigen::MatrixXd &A) {
24 if (A.rows() != A.cols()) {
25 std::cerr << "Error: LU decomposition requires a square matrix, but received a " << A.rows() << "x"
26 << A.cols() << " matrix." << std::endl;
27 throw std::invalid_argument("Matrix must be square for LU decomposition.");
28 }
29
30 n = A.rows();
31 LU = A; // Combined L and U in the same matrix
32 perm.resize(n);
33
34 // Initialize permutation vector
35 for (int i = 0; i < n; ++i) perm[i] = i;
36
37 for (int k = 0; k < n; ++k) {
38 // Partial Pivoting: Find the row with the largest absolute value in the current column
39 int pivotRow = k;
40 double maxVal = std::abs(LU(perm[k], k)); // Compare on the permuted row
41
42 for (int i = k + 1; i < n; ++i) {
43 if (std::abs(LU(perm[i], k)) > maxVal) {
44 maxVal = std::abs(LU(perm[i], k));
45 pivotRow = i;
46 }
47 }
48
49 // Swap rows in the permutation vector if necessary
50 if (pivotRow != k) { std::swap(perm[k], perm[pivotRow]); }
51
52 // Numerical stability check (to prevent division by near-zero values)
53 if (std::abs(LU(perm[k], k)) < 1e-12) { std::cerr << "Warning: Near-zero pivot encountered!" << std::endl; }
54
55 // Perform the elimination (on the original matrix, no perm needed in LU)
56 for (int i = k + 1; i < n; ++i) {
57 LU(perm[i], k) /= LU(perm[k], k); // Use perm[k] for pivot row
58 for (int j = k + 1; j < n; ++j) { LU(perm[i], j) -= LU(perm[i], k) * LU(perm[k], j); }
59 }
60 }
61 }

63 // Solve the system Ax = b using the computed LU decomposition
64 Eigen::VectorXd solve(const Eigen::VectorXd &b) {
65 if (b.size() != n) {
66 std::cerr << "Error: The vector b size does not match the LU matrix dimensions." << std::endl;
67 throw std::invalid_argument("Dimension mismatch between LU matrix and b vector.");
68 }
69
70 Eigen::VectorXd y = Eigen::VectorXd::Zero(n);
71 Eigen::VectorXd x = Eigen::VectorXd::Zero(n);
72
73 // Apply the permutation vector to the right-hand side
74 Eigen::VectorXd b_permuted(n);
75 for (int i = 0; i < n; ++i) { b_permuted(i) = b(perm[i]); }
76
77 // Forward substitution to solve Ly = Pb
78 for (int i = 0; i < n; ++i) {
79 y(i) = b_permuted(i);
80 for (int j = 0; j < i; ++j) {
81 y(i) -= LU(perm[i], j) * y(j); // Use perm[i] to access the permuted row
82 }
83 }
84
85 // Backward substitution to solve Ux = y
86 for (int i = n - 1; i >= 0; --i) {
87 x(i) = y(i);
88 for (int j = i + 1; j < n; ++j) {
89 x(i) -= LU(perm[i], j) * x(j); // Use perm[i] to access the permuted row
90 }
91 x(i) /= LU(perm[i], i);
92 }
93
94 return x;
95 }
