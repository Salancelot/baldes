123template <Direction D>
124void BucketGraph::define_buckets() {
125 int num_intervals = MAIN_RESOURCES;
126 std::vector<double> total_ranges(num_intervals);
127 std::vector<double> base_intervals(num_intervals);
128
129 // Ensure the base_intervals storage is resized for all nodes
130 if constexpr (D == Direction::Forward) {
131 fw_base_intervals.resize(num_intervals);
132 } else {
133 bw_base_intervals.resize(num_intervals);
134 }
135
136 // Determine the base interval and other relevant values for each resource
137 for (int r = 0; r < num_intervals; ++r) {
138 total_ranges[r] = R_max[r] - R_min[r];
139 base_intervals[r] = total_ranges[r] / intervals[r].interval;
140 }
141
142 if constexpr (D == Direction::Forward) {
143 fw_base_intervals = base_intervals;
144 } else {
145 bw_base_intervals = base_intervals;
146 }
147
148 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
149 auto &num_buckets = assign_buckets<D>(num_buckets_fw, num_buckets_bw);
150 auto &num_buckets_index = assign_buckets<D>(num_buckets_index_fw, num_buckets_index_bw);
151 auto &node_interval_trees = assign_buckets<D>(fw_node_interval_trees, bw_node_interval_trees);
152 auto &buckets_size = assign_buckets<D>(fw_buckets_size, bw_buckets_size);
153 num_buckets.resize(nodes.size());
154 num_buckets_index.resize(nodes.size());
155
156 int cum_sum = 0; // Tracks global bucket index
157 int bucket_index = 0;
158
159 node_interval_trees.assign(nodes.size(), SplayTree());
160
161 // Loop through each node to define its specific buckets
162 for (const auto &VRPNode : nodes) {
163 std::vector<double> node_base_interval(num_intervals);
164 for (int r = 0; r < num_intervals; ++r) {
165 node_base_interval[r] = (VRPNode.ub[r] - VRPNode.lb[r]) / intervals[r].interval;
166 }
167 SplayTree node_tree;
168
169 // Multiple buckets case
170 int n_buckets = 0;
171 std::vector<int> current_pos(num_intervals, 0);
172 std::vector<double> interval_start(num_intervals), interval_end(num_intervals);
173
174 // Calculate the start and end for each interval dimension
175 for (int r = 0; r < num_intervals; ++r) {
176 for (auto j = 0; j < intervals[r].interval; ++j) {
177
178 if constexpr (D == Direction::Forward) {
179 interval_start[r] = VRPNode.lb[r] + current_pos[r] * node_base_interval[r];
180
181 if (j == intervals[r].interval - 1) {
182 interval_end[r] = VRPNode.ub[r];
183 } else {
184 interval_end[r] = VRPNode.lb[r] + (current_pos[r] + 1) * node_base_interval[r];
185 }
186 } else {
187 if (j == intervals[r].interval - 1) {
188 interval_start[r] = VRPNode.lb[r];
189 } else {
190 interval_start[r] = VRPNode.ub[r] - (current_pos[r] + 1) * node_base_interval[r];
191 }
192 //interval_start[r] = VRPNode.ub[r] - (current_pos[r] + 1) * node_base_interval[r];
193 interval_end[r] = VRPNode.ub[r] - current_pos[r] * node_base_interval[r];
194 }
195
196 // Apply rounding to two decimal places before using values
197 interval_start[r] = roundToTwoDecimalPlaces(interval_start[r]);
198 interval_end[r] = roundToTwoDecimalPlaces(interval_end[r]);
199
200 if constexpr (D == Direction::Backward) {
201 interval_start[r] = std::max(interval_start[r], R_min[r]);
202 } else {
203 interval_end[r] = std::min(interval_end[r], R_max[r]);
204 }
205
206 buckets.push_back(Bucket(VRPNode.id, interval_start, interval_end));
207
208 node_tree.insert(interval_start, interval_end, bucket_index);
209
210 bucket_index++;
211 n_buckets++;
212 cum_sum++;
213
214 current_pos[r]++;
215 }
216 }
217
218 // Update node-specific bucket data
219 num_buckets[VRPNode.id] = n_buckets;
220 num_buckets_index[VRPNode.id] = cum_sum - n_buckets;
221
222 node_interval_trees[VRPNode.id] = node_tree;
223 }
224
225 // Update global bucket sizes based on direction
226 buckets_size = cum_sum;
227}

235template <Direction D>
236void BucketGraph::generate_arcs() {
237 // Mutex to synchronize access to shared resources
238 auto buckets_mutex = std::mutex();
239
240 // Clear the appropriate bucket graph (forward or backward)
241 if constexpr (D == Direction::Forward) {
242 fw_bucket_graph.clear(); // Clear the forward bucket graph
243 } else {
244 bw_bucket_graph.clear(); // Clear the backward bucket graph
245 }
246
247 // Assign the forward or backward fixed buckets and other bucket-related structures
248 auto &fixed_buckets = assign_buckets<D>(fw_fixed_buckets, bw_fixed_buckets);
249 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
250 auto &num_buckets = assign_buckets<D>(num_buckets_fw, num_buckets_bw);
251 auto &num_buckets_index = assign_buckets<D>(num_buckets_index_fw, num_buckets_index_bw);
252
253 // Clear all buckets in parallel, removing any existing arcs
254 std::for_each(buckets.begin(), buckets.end(), [&](auto &bucket) {
255 bucket.clear(); // Clear bucket data
256 bucket.clear_arcs(D == Direction::Forward); // Clear arcs in the bucket
257 });
258
259 auto add_arcs_for_node = [&](const VRPNode &node, int from_bucket, std::vector<double> &res_inc,
260 std::vector<std::pair<int, int>> &local_arcs) {
261 // Retrieve the arcs for the node in the given direction (Forward/Backward)
262 auto arcs = node.get_arcs<D>();
263 // Compute base intervals for each resource dimension based on R_max and R_min
264 std::vector<double> node_intervals(intervals.size());
265 for (int r = 0; r < intervals.size(); ++r) {
266 node_intervals[r] = (node.ub[r] - node.lb[r]) / intervals[r].interval;
267 }
268
269 // Iterate over all arcs of the node
270 for (const auto &arc : arcs) {
271 const auto &next_node = nodes[arc.to]; // Get the destination node of the arc
272
273 std::vector<double> next_node_intervals(intervals.size());
274 for (int r = 0; r < intervals.size(); ++r) {
275 next_node_intervals[r] = (next_node.ub[r] - next_node.lb[r]) / intervals[r].interval;
276 }
277 // Skip self-loops (no arc from a node to itself)
278 if (node.id == next_node.id) continue;
279
280 // Calculate travel cost and cost increment based on node's properties
281 const auto travel_cost = getcij(node.id, next_node.id);
282 double cost_inc = travel_cost - next_node.cost;
283 res_inc[0] = travel_cost + node.duration; // Update resource increment based on node duration
284 // Iterate over all possible destination buckets for the next node
285
286 for (int j = 0; j < num_buckets[next_node.id]; ++j) {
287 int to_bucket = j + num_buckets_index[next_node.id];
288 if (from_bucket == to_bucket) continue; // Skip arcs that loop back to the same bucket
289
290 if (fixed_buckets[from_bucket][to_bucket] == 1) continue; // Skip fixed arcs
291
292 bool valid_arc = true;
293 for (int r = 0; r < res_inc.size(); ++r) {
294 // Forward direction: Check that resource increment doesn't exceed upper bounds
295 if constexpr (D == Direction::Forward) {
296 if (buckets[from_bucket].lb[r] + res_inc[r] > next_node.ub[r]) {
297 valid_arc = false;
298 break;
299 }
300 }
301 // Backward direction: Check that resource decrement doesn't drop below lower bounds
302 else if constexpr (D == Direction::Backward) {
303 if (buckets[from_bucket].ub[r] - res_inc[r] < next_node.lb[r]) {
304 valid_arc = false;
305 break;
306 }
307 }
308 }
309 if (!valid_arc) continue; // Skip invalid arcs
310
311 // Further refine arc validity based on the base intervals and node bounds
312 if constexpr (D == Direction::Forward) {
313 for (int r = 0; r < res_inc.size(); ++r) {
314 double max_calc = std::max(buckets[from_bucket].lb[r] + res_inc[r], next_node.lb[r]);
315 if (max_calc < buckets[to_bucket].lb[r] ||
316 max_calc >= buckets[to_bucket].lb[r] + next_node_intervals[r] + numericutils::eps) {
317 valid_arc = false;
318 break;
319 }
320 }
321 } else if constexpr (D == Direction::Backward) {
322 for (int r = 0; r < res_inc.size(); ++r) {
323 double min_calc = std::min(buckets[from_bucket].ub[r] - res_inc[r], next_node.ub[r]);
324 if (min_calc > buckets[to_bucket].ub[r] ||
325 min_calc <= buckets[to_bucket].ub[r] - next_node_intervals[r] - numericutils::eps) {
326 valid_arc = false;
327 break;
328 }
329 }
330 }
331 if (!valid_arc) continue; // Skip invalid arcs
332
333 // Store the arc data locally before committing it to the global structure
334 local_arcs.emplace_back(from_bucket, to_bucket);
335 double local_cost_inc = cost_inc;
336 std::vector<double> local_res_inc = res_inc;
337
338 // Add the arc to the global structure and the bucket
339 {
340 std::lock_guard<std::mutex> lock(buckets_mutex); // Lock the mutex to ensure thread safety
341 add_arc<D>(from_bucket, to_bucket, local_res_inc, local_cost_inc); // Add the arc globally
342 buckets[from_bucket].add_bucket_arc(from_bucket, to_bucket, local_res_inc, local_cost_inc,
343 D == Direction::Forward,
344 false); // Add the arc to the bucket
345 }
346 }
347 }
348 };
349
350 unsigned int total_threads = std::thread::hardware_concurrency();
351 // Divide by 2 to use only half of the available threads
352 unsigned int half_threads = total_threads / 2;
353
354 const int JOBS = half_threads;
355 exec::static_thread_pool pool(JOBS);
356 auto sched = pool.get_scheduler();
357
358 // Iterate over all nodes in parallel, generating arcs for each
359 std::vector<int> tasks; // Tasks will store node ids
360 for (int node_id = 0; node_id < nodes.size(); ++node_id) {
361 tasks.push_back(node_id); // Store the node id to be processed
362 }
363
364 // Define chunk size to reduce parallelization overhead
365 const int chunk_size = 10; // Adjust based on your performance needs
366
367 // Parallel execution in chunks
368 auto bulk_sender = stdexec::bulk(
369 stdexec::just(), (tasks.size() + chunk_size - 1) / chunk_size,
370 [this, &tasks, &num_buckets, &num_buckets_index, &add_arcs_for_node, chunk_size](std::size_t chunk_idx) {
371 size_t start_idx = chunk_idx * chunk_size;
372 size_t end_idx = std::min(start_idx + chunk_size, tasks.size());
373
374 // Process a chunk of tasks (i.e., a group of nodes)
375 for (size_t task_idx = start_idx; task_idx < end_idx; ++task_idx) {
376 int node_id = tasks[task_idx]; // Get the node id
377 const VRPNode &VRPNode = nodes[node_id];
378
379 std::vector<double> res_inc = {static_cast<double>(VRPNode.duration)}; // Resource increment vector
380 std::vector<std::pair<int, int>> local_arcs; // Local storage for arcs
381
382 // Generate arcs for all buckets associated with the current node
383 for (int i = 0; i < num_buckets[VRPNode.id]; ++i) {
384 int from_bucket = i + num_buckets_index[VRPNode.id]; // Determine the source bucket
385 add_arcs_for_node(VRPNode, from_bucket, res_inc,
386 local_arcs); // Add arcs for this node and bucket
387 }
388 }
389 });
390
391 // Submit work to the thread pool
392 auto work = stdexec::starts_on(sched, bulk_sender);
393 stdexec::sync_wait(std::move(work));
394}

405template <Direction D>
406Label *BucketGraph::get_best_label(const std::vector<int> &topological_order, const std::vector<double> &c_bar,
407 const std::vector<std::vector<int>> &sccs) {
408 double best_cost = std::numeric_limits<double>::infinity();
409 Label *best_label = nullptr; // Ensure this is initialized
410 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
411
412 for (const int component_index : topological_order) {
413 const auto &component_buckets = sccs[component_index];
414
415 for (const int bucket : component_buckets) {
416 const auto &label = buckets[bucket].get_best_label();
417 if (!label) continue;
418
419 if (label->cost < best_cost) {
420 best_cost = label->cost;
421 best_label = label;
422 }
423 }
424 }
425
426 return best_label;
427}

433template <Stage S>
434void BucketGraph::ConcatenateLabel(const Label *L, int &b, Label *&pbest, std::vector<uint64_t> &Bvisited) {
435 // Use a vector for iterative processing as a stack
436 std::vector<int> bucket_stack;
437 bucket_stack.reserve(50);
438 bucket_stack.push_back(b);
439
440 const auto &L_node_id = L->node_id;
441 const auto &L_resources = L->resources;
442 const auto &L_last_node = nodes[L_node_id];
443
444 while (!bucket_stack.empty()) {
445 // Pop the next bucket from the stack (vector back)
446 int current_bucket = bucket_stack.back();
447 bucket_stack.pop_back();
448
449 // Mark the bucket as visited
450 const size_t segment = current_bucket >> 6; // Equivalent to current_bucket / 64
451 const size_t bit_position = current_bucket & 63; // Equivalent to current_bucket % 64
452
453 Bvisited[segment] |= (1ULL << bit_position);
454
455 const auto &bucketLprimenode = bw_buckets[current_bucket].node_id;
456 double cost = getcij(L_node_id, bucketLprimenode);
457
458#if defined(RCC) || defined(EXACT_RCC)
459 if constexpr (S == Stage::Four) { cost -= arc_duals.getDual(L_node_id, bucketLprimenode); }
460#endif
461
462 // Branching duals
463 if (branching_duals->size() > 0) { cost -= branching_duals->getDual(L_node_id, bucketLprimenode); }
464
465#ifdef SRC
466 decltype(cut_storage) cutter = nullptr;
467 decltype(cut_storage->SRCDuals) *SRCDuals = nullptr;
468
469 if constexpr (S > Stage::Three) {
470 cutter = cut_storage; // Initialize cutter
471 SRCDuals = &cutter->SRCDuals; // Initialize SRCDuals
472 }
473#endif
474
475 double L_cost_plus_cost = L->cost + cost;
476
477 // Early exit based on cost comparison
478 if ((S != Stage::Enumerate && L_cost_plus_cost + bw_c_bar[current_bucket] >= pbest->cost) ||
479 (S == Stage::Enumerate && L_cost_plus_cost + bw_c_bar[current_bucket] >= gap)) {
480 continue;
481 }
482
483 // Get the labels for the current bucket
484 auto &bucket = bw_buckets[current_bucket];
485 const auto &labels = bucket.get_labels();
486
487 for (const auto &L_bw : labels) {
488 if (L_bw->node_id == L_node_id || !check_feasibility(L, L_bw)) continue;
489 double candidate_cost = L_cost_plus_cost + L_bw->cost;
490
491#ifdef SRC
492 if constexpr (S > Stage::Three) {
493 for (auto it = cutter->begin(); it < cutter->end(); ++it) {
494 if ((*SRCDuals)[it->id] == 0) continue;
495 auto den = it->p.den;
496 auto sum = (L->SRCmap[it->id] + L_bw->SRCmap[it->id]);
497 if (sum >= den) { candidate_cost -= (*SRCDuals)[it->id]; }
498 }
499 }
500#endif
501
502 // Check for visited overlap and skip if true
503
504 if constexpr (S >= Stage::Three) {
505 bool visited_overlap = false;
506 for (size_t i = 0; i < L->visited_bitmap.size(); ++i) {
507 if (L->visited_bitmap[i] & L_bw->visited_bitmap[i]) {
508 visited_overlap = true;
509 break;
510 }
511 }
512 if (visited_overlap) continue;
513 }
514
515 // Early exit based on candidate cost
516 if ((S != Stage::Enumerate && candidate_cost >= pbest->cost) ||
517 (S == Stage::Enumerate && candidate_cost >= gap)) {
518 continue;
519 }
520
521 // Compute and store the new label
522 pbest = compute_label(L, L_bw);
523 merged_labels.push_back(pbest);
524 }
525
526 // Add unvisited neighboring buckets to the stack (vector back)
527 for (int b_prime : Phi_bw[current_bucket]) {
528 const size_t segment_prime = b_prime >> 6;
529 const size_t bit_position_prime = b_prime & 63;
530 if (!(Bvisited[segment_prime] & (1ULL << bit_position_prime))) {
531 bucket_stack.push_back(b_prime); // Add to the vector stack
532 }
533 }
534 }
535}

546template <Direction D>
547void BucketGraph::SCC_handler() {
548 auto &Phi = assign_buckets<D>(Phi_fw, Phi_bw);
549 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
550 auto &bucket_graph = assign_buckets<D>(fw_bucket_graph, bw_bucket_graph);
551 ankerl::unordered_dense::map<int, std::vector<int>> extended_bucket_graph = bucket_graph;
552
553 // Extend the bucket graph with arcs defined by the Phi sets
554 for (auto i = 0; i < extended_bucket_graph.size(); ++i) {
555 auto phi_set = Phi[i];
556 if (phi_set.empty()) continue;
557 for (auto &phi_bucket : phi_set) { extended_bucket_graph[phi_bucket].push_back(i); }
558 }
559
560 SCC scc_finder;
561 scc_finder.convertFromUnorderedMap(extended_bucket_graph); // print extended bucket graph
562
563 auto sccs = scc_finder.tarjanSCC();
564 auto topological_order = scc_finder.topologicalOrderOfSCCs(sccs);
565
566#ifdef VERBOSE
567 // print SCCs and buckets in it
568 constexpr auto blue = "\033[34m";
569 constexpr auto reset = "\033[0m";
570 if constexpr (D == Direction::Forward) {
571 fmt::print("FW SCCs: \n");
572 } else {
573 fmt::print("BW SCCs: \n");
574 }
575 for (auto scc : topological_order) {
576 if constexpr (D == Direction::Forward) {
577 fmt::print("{}({}) -> {}", blue, scc, reset);
578 } else {
579 fmt::print("{}({}) -> {}", blue, scc, reset);
580 }
581 for (auto &bucket : sccs[scc]) { fmt::print("{} ", bucket); }
582 }
583 fmt::print("\n");
584#endif
585
586 std::vector<std::vector<int>> ordered_sccs;
587 ordered_sccs.reserve(sccs.size()); // Reserve space for all SCCs
588 for (int i : topological_order) { ordered_sccs.push_back(sccs[i]); }
589
590 auto sorted_sccs = sccs;
591 for (auto &scc : sorted_sccs) {
592 if constexpr (D == Direction::Forward) {
593 std::sort(scc.begin(), scc.end(), [&buckets](int a, int b) { return buckets[a].lb[0] < buckets[b].lb[0]; });
594 } else {
595 std::sort(scc.begin(), scc.end(), [&buckets](int a, int b) { return buckets[a].ub[0] > buckets[b].ub[0]; });
596 }
597 }
598
599 // iterate over nodes
600 for (auto &node : nodes) {
601 if constexpr (D == Direction::Forward) {
602 node.fw_arcs_scc.resize(sccs.size());
603 } else {
604 node.bw_arcs_scc.resize(sccs.size());
605 }
606 }
607
608 // Split arcs for each SCC
609 auto scc_ctr = 0;
610 for (const auto &scc : sccs) {
611 // Iterate over each bucket in the SCC
612 for (int bucket : scc) {
613 int from_node_id = buckets[bucket].node_id; // Get the source node ID
614 VRPNode &node = nodes[from_node_id]; // Access the corresponding node
615 // Define filtered arcs depending on the direction
616 if constexpr (D == Direction::Forward) {
617 std::vector<Arc> &filtered_fw_arcs = nodes[from_node_id].fw_arcs_scc[scc_ctr]; // For forward direction
618
619 // Iterate over the arcs from the current bucket
620 const auto &bucket_arcs = buckets[bucket].template get_bucket_arcs<D>();
621 for (const auto &arc : bucket_arcs) {
622 int to_node_id = buckets[arc.to_bucket].node_id; // Get the destination node ID
623
624 // Search for the arc from `from_node_id` to `to_node_id` in the node's arcs
625 auto it = std::find_if(node.fw_arcs.begin(), node.fw_arcs.end(),
626 [&to_node_id](const Arc &a) { return a.to == to_node_id; });
627
628 // If both nodes are within the current SCC, retain the arc
629 if (it != node.fw_arcs.end()) {
630 // Add the arc to the filtered arcs
631 filtered_fw_arcs.push_back(*it); // Forward arcs
632 }
633 }
634 } else {
635 std::vector<Arc> &filtered_bw_arcs = nodes[from_node_id].bw_arcs_scc[scc_ctr]; // For forward direction
636
637 // Iterate over the arcs from the current bucket
638 const auto &bucket_arcs = buckets[bucket].template get_bucket_arcs<D>();
639 for (const auto &arc : bucket_arcs) {
640 int to_node_id = buckets[arc.to_bucket].node_id; // Get the destination node ID
641
642 // Search for the arc from `from_node_id` to `to_node_id` in the node's arcs
643 auto it = std::find_if(node.bw_arcs.begin(), node.bw_arcs.end(),
644 [&to_node_id](const Arc &a) { return a.to == to_node_id; });
645
646 // If both nodes are within the current SCC, retain the arc
647 if (it != node.bw_arcs.end()) {
648 // Add the arc to the filtered arcs
649 filtered_bw_arcs.push_back(*it); // Forward arcs
650 }
651 }
652 }
653 }
654
655 // Increment SCC counter
656 ++scc_ctr;
657 }
658
659 for (auto &node : nodes) {
660 if constexpr (D == Direction::Forward) {
661 // Iterate over all SCCs for each node
662 for (auto &fw_arcs_scc : node.fw_arcs_scc) {
663 // Sort arcs by from_bucket and to_bucket
664 std::sort(fw_arcs_scc.begin(), fw_arcs_scc.end(),
665 [](const Arc &a, const Arc &b) { return std::tie(a.from, a.to) < std::tie(b.from, b.to); });
666
667 // Remove consecutive duplicates
668 auto last = std::unique(fw_arcs_scc.begin(), fw_arcs_scc.end(),
669 [](const Arc &a, const Arc &b) { return a.from == b.from && a.to == b.to; });
670
671 // Erase the duplicates from the vector
672 fw_arcs_scc.erase(last, fw_arcs_scc.end());
673 }
674 } else {
675 // Iterate over all SCCs for each node
676 for (auto &bw_arcs_scc : node.bw_arcs_scc) {
677 // Sort arcs by from_bucket and to_bucket
678 std::sort(bw_arcs_scc.begin(), bw_arcs_scc.end(),
679 [](const Arc &a, const Arc &b) { return std::tie(a.from, a.to) < std::tie(b.from, b.to); });
680
681 // Remove consecutive duplicates
682 auto last = std::unique(bw_arcs_scc.begin(), bw_arcs_scc.end(),
683 [](const Arc &a, const Arc &b) { return a.from == b.from && a.to == b.to; });
684
685 // Erase the duplicates from the vector
686 bw_arcs_scc.erase(last, bw_arcs_scc.end());
687 }
688 }
689 }
690
691 UnionFind uf(ordered_sccs);
692 if constexpr (D == Direction::Forward) {
693 fw_ordered_sccs = ordered_sccs;
694 fw_topological_order = topological_order;
695 fw_sccs = sccs;
696 fw_sccs_sorted = sorted_sccs;
697 fw_union_find = uf;
698 } else {
699 bw_ordered_sccs = ordered_sccs;
700 bw_topological_order = topological_order;
701 bw_sccs = sccs;
702 bw_sccs_sorted = sorted_sccs;
703 bw_union_find = uf;
704 }

807template <Stage S>
808void BucketGraph::heuristic_fixing() {
809 // Stage 3 fixing heuristic
810 reset_pool();
811 reset_fixed();
812 common_initialization();
813
814 std::vector<double> forward_cbar(fw_buckets.size(), std::numeric_limits<double>::infinity());
815 std::vector<double> backward_cbar(bw_buckets.size(), std::numeric_limits<double>::infinity());
816
817 run_labeling_algorithms<Stage::Two, Full::Partial>(forward_cbar, backward_cbar);
818
819 std::vector<std::vector<Label *>> fw_labels_map(nodes.size());
820 std::vector<std::vector<Label *>> bw_labels_map(nodes.size());
821
822 auto group_labels = [&](auto &buckets, auto &labels_map) {
823 for (auto &bucket : buckets) {
824 for (auto &label : bucket.get_labels()) {
825 labels_map[label->node_id].push_back(label); // Directly index using node_id
826 }
827 }
828 };
829
830 // Create tasks for forward and backward labels grouping
831 auto forward_task = stdexec::schedule(bi_sched) | stdexec::then([&]() { group_labels(fw_buckets, fw_labels_map); });
832 auto backward_task =
833 stdexec::schedule(bi_sched) | stdexec::then([&]() { group_labels(bw_buckets, bw_labels_map); });
834
835 // Execute the tasks in parallel
836 auto work = stdexec::when_all(std::move(forward_task), std::move(backward_task));
837
838 stdexec::sync_wait(std::move(work));
839
840 auto num_fixes = 0;
841 // Function to find the minimum cost label in a vector of labels
842 auto find_min_cost_label = [](const std::vector<Label *> &labels) -> const Label * {
843 return *std::min_element(labels.begin(), labels.end(),
844 [](const Label *a, const Label *b) { return a->cost < b->cost; });
845 };
846 for (const auto &node_I : nodes) {
847 const auto &fw_labels = fw_labels_map[node_I.id];
848 if (fw_labels.empty()) continue; // Skip if no labels for this node_id
849
850 for (const auto &node_J : nodes) {
851 if (node_I.id == node_J.id) continue; // Compare based on id (or other key field)
852 const auto &bw_labels = bw_labels_map[node_J.id];
853 if (bw_labels.empty()) continue; // Skip if no labels for this node_id
854
855 const Label *min_fw_label = find_min_cost_label(fw_labels);
856 const Label *min_bw_label = find_min_cost_label(bw_labels);
857
858 if (!min_fw_label || !min_bw_label) continue;
859
860 const VRPNode &L_last_node = nodes[min_fw_label->node_id];
861 auto cost = getcij(min_fw_label->node_id, min_bw_label->node_id);
862
863 // Check for infeasibility
864 if (min_fw_label->resources[TIME_INDEX] + cost + L_last_node.consumption[TIME_INDEX] >
865 min_bw_label->resources[TIME_INDEX]) {
866 continue;
867 }
868
869 if (min_fw_label->cost + cost + min_bw_label->cost > gap) {
870 fixed_arcs[node_I.id][node_J.id] = 1; // Index with node ids
871 num_fixes++;
872 }
873 }
874 }
