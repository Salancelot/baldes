133 // Erase the cut from the cuts vector
134 cuts.erase(cuts.begin() + cutIndex);

136 for (auto &entry : cuts) {

138 entry.id--; // Decrement index to account for the removed cut

145
146 if (it != cutMaster_to_cut_map.end()) {
147 // Remove the entry for the removed cut

219 auto computeLimitedMemoryCoefficients(const std::vector<int> &P) {
220 // iterate over cuts
221 std::vector<double> alphas;
222 alphas.reserve(cuts.size());
223 for (auto c : cuts) {
224 double alpha = 0.0;
225 int S = 0;
226 auto AM = c.neighbors;
227 auto C = c.baseSet;
228 auto p = c.p;
229 auto order = c.baseSetOrder;
230
231 for (size_t j = 1; j < P.size() - 1; ++j) {
232 int vj = P[j];
233
234 // Check if the node vj is in AM (bitwise check)
235 if (!(AM[vj / 64] & (1ULL << (vj % 64)))) {
236 S = 0; // Reset S if vj is not in AM
237 }
238 if (C[vj / 64] & (1ULL << (vj % 64))) {
239 // Get the position of vj in C by counting the set bits up to vj
240 int pos = order[vj];
241 S += p.num[pos];
242 if (S % p.den == 0) {
243 // S -= 1;
244 alpha += 1;
245 }
246 }
247 }
248
249 alphas.push_back(alpha);
250 }
251 return alphas;
252 }
