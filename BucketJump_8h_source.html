36void BucketGraph::UpdateBucketsSet(const double theta, const Label *label, ankerl::unordered_dense::set<int> &Bbidi,
37 int &current_bucket, std::vector<uint64_t> &Bvisited) {
38 // Precompute values and assign references for the direction-specific data
39
40 auto &Phi_opposite = assign_buckets<D>(Phi_bw, Phi_fw);
41 auto &buckets_opposite = assign_buckets<D>(bw_buckets, fw_buckets);
42 auto &c_bar_opposite = assign_buckets<D>(bw_c_bar, fw_c_bar);
43
44 const int bucketLnode = label->node_id;
45 std::vector<int> bucket_stack;
46 bucket_stack.reserve(10); // Pre-allocate space for the stack
47 bucket_stack.push_back(current_bucket);
48
49 // Lambda to compare bitmaps
50 auto bitmaps_conflict = [&](const Label *L1, const Label *L2) {
51 for (size_t i = 0; i < L1->visited_bitmap.size(); ++i) {
52 if ((L1->visited_bitmap[i] & L2->visited_bitmap[i]) != 0) { return true; }
53 }
54 return false;
55 };
56
57 while (!bucket_stack.empty()) {
58 int curr_bucket = bucket_stack.back();
59 bucket_stack.pop_back();
60 const size_t segment = curr_bucket >> 6; // Determine the segment for the current bucket
61 const size_t bit_position = curr_bucket & 63; // Determine the bit position within the segment
62 Bvisited[segment] |= (1ULL << bit_position); // Set the bit corresponding to the current bucket as visited
63 // print current bucket
64 const int bucketLprimenode = buckets_opposite[curr_bucket].node_id;
65 double cost = getcij(bucketLnode, bucketLprimenode);
66
67#ifdef RCC
68 auto arc_dual = arc_duals.getDual(bucketLnode, bucketLprimenode);
69 cost -= arc_dual;
70#endif
71 // Stop exploring if the cost exceeds the threshold
72 if (label->cost + cost + c_bar_opposite[curr_bucket] >= theta) { continue; }
73
74 // Only process if bucket is not already in Bbidi
75 if (Bbidi.insert(curr_bucket).second) {
76 const auto &opposite_labels = buckets_opposite[curr_bucket].get_labels();
77 for (const auto &L : opposite_labels) {
78 if (label->node_id == L->node_id || bitmaps_conflict(label, L)) {
79 continue; // Skip labels with the same node ID or conflicting bitmaps
80 }
81
82 const VRPNode &L_last_node = (D == Direction::Forward) ? nodes[label->node_id] : nodes[L->node_id];
83 const double time_constraint = (D == Direction::Forward)
84 ? label->resources[TIME_INDEX] + cost + L_last_node.duration
85 : L->resources[TIME_INDEX] + cost + L_last_node.duration;
86
87 if (time_constraint >
88 ((D == Direction::Forward) ? L->resources[TIME_INDEX] : label->resources[TIME_INDEX])) {
89 continue; // Skip if time constraint is violated
90 }
91
92 // Insert the current bucket into Bbidi if the total cost is below theta
93 if (label->cost + cost + L->cost < theta) { Bbidi.emplace(curr_bucket); }
94 }
95 }
96
97 // Add neighboring buckets to the stack, only if not visited
98 for (int b_prime : Phi_opposite[curr_bucket]) {
99 const size_t segment_prime = b_prime >> 6; // Determine the segment for the neighboring bucket
100 const size_t bit_position_prime = b_prime & 63; // Determine the bit position within the segment
101
102 // If the neighboring bucket hasn't been visited, push it onto the stack
103 if ((Bvisited[segment_prime] & (1ULL << bit_position_prime)) == 0) { bucket_stack.push_back(b_prime); }
104 }
105 }
106}

129 using ArcMap = ankerl::unordered_dense::map<std::pair<std::pair<int, int>, int>, ankerl::unordered_dense::set<int>,
130 arc_map_hash>;

132 ArcMap local_B_Ba_b;

137 // Function to process jump arcs
138 auto process_jump_arcs = [&](int b) {
139 const auto &jump_arcs = buckets[b].template get_jump_arcs<D>();
140 if (!jump_arcs.empty()) {
141 const size_t n_segments = buckets_size / 64 + 1;
142 std::vector<uint64_t> Bvisited(n_segments, 0);
143 auto &labels = buckets[b].get_labels();
144 for (const auto &a : jump_arcs) {
145 auto increment = a.resource_increment;

152
153 for (auto &L_item : labels) {
154 Bidi_map.insert(b);
155 std::memset(Bvisited.data(), 0, Bvisited.size() * sizeof(uint64_t));
156 UpdateBucketsSet<D>(theta, L_item, Bidi_map, b_opposite, Bvisited);
157 }
158 }
159 }
160 };

162 // Function to process bucket arcs
163 auto process_bucket_arcs = [&](int b) {
164 const auto &bucket_arcs = buckets[b].template get_bucket_arcs<D>();
165 auto &labels = buckets[b].get_labels();
166
167 const size_t n_segments = buckets_size / 64 + 1;
168 std::vector<uint64_t> Bvisited(n_segments, 0);
169
170 for (const auto &a : bucket_arcs) {
171 std::vector<double> increment(MAIN_RESOURCES, 0);
172 for (int r = 0; r < MAIN_RESOURCES; ++r) {
173 if constexpr (D == Direction::Forward) {
174 increment[r] = buckets[b].lb[r] + a.resource_increment[r];
175 } else {
176 increment[r] = buckets[b].ub[r] - a.resource_increment[r];
177 }
178 }
179 // print b
180 auto arc_key = create_arc_key(buckets[a.from_bucket].node_id, buckets[a.to_bucket].node_id, b);
181 int b_opposite = get_opposite_bucket_number<D>(a.to_bucket, increment);
182 // print b_opposite
183 auto &Bidi_map = local_B_Ba_b[arc_key];
184
185 for (auto &L_item : labels) {
186 Bidi_map.insert(b);
187 std::memset(Bvisited.data(), 0, Bvisited.size() * sizeof(uint64_t));
188 UpdateBucketsSet<D>(theta, L_item, Bidi_map, b_opposite, Bvisited);
189 }
190
191 // Check for opposite direction arcs
192 if (auto it = local_B_Ba_b.find(arc_key); it != local_B_Ba_b.end()) {
193 auto &Bidi_map_opposite = it->second;
194 bool contains =
195 std::any_of(Phi_opposite[b_opposite].begin(), Phi_opposite[b_opposite].end(), [&](int b_prime) {
196 return Bidi_map_opposite.find(b_prime) != Bidi_map_opposite.end();
197 });
198
199 if (!contains && Bidi_map_opposite.find(b_opposite) == Bidi_map_opposite.end()) {
200 fixed_buckets[a.from_bucket][a.to_bucket] = 1;
201 ++removed_arcs;
202 }
203 }
204 }
205 };
206
207 // Process each bucket
208 for (int b = 0; b < buckets_size; ++b) {
209 const auto &node_arcs = nodes[buckets[b].node_id].template get_arcs<D>();
210
211 // Process node arcs
212 for (const auto &a : node_arcs) {
213 auto arc_key = std::make_pair(std::make_pair(a.from, a.to), b);
214 auto &Bidi_map = local_B_Ba_b[arc_key];
215
216 if (!Phi[b].empty()) {
217 Bidi_map.insert(b);
218 for (const auto &b_prime : Phi[b]) {
219 auto neighbor_key = std::make_pair(std::make_pair(a.from, a.to), b_prime);
220 if (auto it = local_B_Ba_b.find(neighbor_key); it != local_B_Ba_b.end()) {
221 Bidi_map.insert(it->second.begin(), it->second.end());
222 }
223 }
224 }
225 }
226
227 // Process jump and bucket arcs using separate functions
228 process_jump_arcs(b);
229 process_bucket_arcs(b);
230 }
231
232 // Print the number of arcs removed
233 if constexpr (D == Direction::Forward) {
234 print_info("[Fw] removed arcs: {}\n", removed_arcs);
235 } else {
236 print_info("[Bw] removed arcs: {}\n", removed_arcs);
237 }

250template <Direction D>
251void BucketGraph::ObtainJumpBucketArcs() {
252 // Assign forward or backward buckets, fixed buckets, bucket indices, and Phi (adjacency list) based on
253 // direction
254 auto &buckets = assign_buckets<D>(fw_buckets, bw_buckets);
255 auto &fixed_buckets = assign_buckets<D>(fw_fixed_buckets, bw_fixed_buckets);
256 auto &num_buckets_index = assign_buckets<D>(num_buckets_index_fw, num_buckets_index_bw);
257 auto &num_buckets = assign_buckets<D>(num_buckets_fw, num_buckets_bw);
258 auto &Phi = assign_buckets<D>(Phi_fw, Phi_bw);
259 auto &buckets_size = assign_buckets<D>(fw_buckets_size, bw_buckets_size);
260
261 int arc_counter = 0; // Counter for jump arcs added
262 int missing_counter = 0; // Counter for missing paths
263
264 // Loop through all buckets
265 for (int b = 0; b < buckets_size; ++b) {
266 std::vector<int> B_bar; // Temporary storage for valid bucket indices
267
268 // Cache the current bucket's node ID and arcs
269 const int current_node_id = buckets[b].node_id;
270 const auto &arcs = buckets[b].template get_bucket_arcs<D>();
271 const auto &original_arcs = nodes[current_node_id].template get_arcs<D>();
272
273 if (arcs.empty()) { continue; } // Skip if no arcs in the current bucket
274
275 // Process each original arc for the current node
276 for (const auto &orig_arc : original_arcs) {
277 const int from_node = orig_arc.from;
278 const int to_node = orig_arc.to;
279 bool have_path = false;
280
281 // Check if the path exists in the current bucket arcs
282 for (const auto &gamma : arcs) {
283 if (fixed_buckets[gamma.from_bucket][gamma.to_bucket] == 1) { continue; }
284
285 const int from_node_b = buckets[gamma.from_bucket].node_id;
286 const int to_node_b = buckets[gamma.to_bucket].node_id;
287
288 if (from_node == from_node_b && to_node == to_node_b) {
289 have_path = true;
290 break; // Path found, exit early
291 }
292 }
293
294 // If no path was found, find a jump arc
295 if (!have_path) {
296 ++missing_counter; // Increment missing path counter
297
298 // Cache the starting bucket and the number of buckets for this node
299 const int start_bucket = num_buckets_index[current_node_id];
300 const int node_buckets = num_buckets[current_node_id];
301
302 auto process_bucket = [&](int b, int b_prime) {
303 // Check if b_prime is adjacent to b in Phi
304 const auto &phi_b_prime = Phi[b_prime];
305 if (std::find(phi_b_prime.begin(), phi_b_prime.end(), b) == phi_b_prime.end()) {
306 return; // Not adjacent, skip
307 }
308
309 // Check arcs in the adjacent bucket b_prime
310 const auto &arcs_prime = buckets[b_prime].template get_bucket_arcs<D>();
311 for (const auto &gamma_prime : arcs_prime) {
312 if (fixed_buckets[gamma_prime.from_bucket][gamma_prime.to_bucket] == 1) { continue; }
313
314 const int from_node_prime = buckets[gamma_prime.from_bucket].node_id;
315 const int to_node_prime = buckets[gamma_prime.to_bucket].node_id;
316
317 // If the arc matches, add a jump arc
318 if (from_node == from_node_prime && to_node == to_node_prime) {
319 B_bar.push_back(b_prime); // Add the valid bucket to B_bar
320
321 // Wrap the resource increment in a vector
322 const std::vector<double> res = gamma_prime.resource_increment;
323 const double cost = gamma_prime.cost_increment;
324
325 buckets[b].add_jump_arc(b, b_prime, res, cost, true);
326 ++arc_counter; // Increment the jump arc counter
327 }
328 }
329 };
330
331 if constexpr (D == Direction::Forward) {
332 // Forward direction: incrementing buckets
333 for (int b_prime = b + 1; b_prime < start_bucket + node_buckets; ++b_prime) {
334 process_bucket(b, b_prime);
335 }
336 } else {
337 // Backward direction: decrementing buckets
338 // for (int b_prime = b - 1; b_prime >= start_bucket; --b_prime) { process_bucket(b, b_prime); }
339 for (int b_prime = b + 1; b_prime < start_bucket + node_buckets; ++b_prime) {
340 process_bucket(b, b_prime);
341 }
342 }
343 }
344 }
345 }
346 CONDITIONAL(D, print_info("[Fw] added {} jump arcs\n", arc_counter),
347 print_info("[Bw] added {} jump arcs\n", arc_counter));
348}
