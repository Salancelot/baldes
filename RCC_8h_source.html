57 std::vector<double> computeRCCCoefficients(const std::vector<int> &label) {
58 std::vector<double> coeffs(cuts_.size(), 0);
59
60 // Precompute the pairs of consecutive nodes in the label
61 ankerl::unordered_dense::set<std::pair<int, int>> label_arcs;
62 for (int j = 0; j < label.size() - 1; j++) { label_arcs.insert({label[j], label[j + 1]}); }
63
64 // Iterate over cuts
65 for (int i = 0, n_cuts = cuts_.size(); i < n_cuts; i++) {
66 int coeff = 0;
67
68 // Iterate over arcs in the cut and check if they exist in the precomputed label arcs
69 for (const auto &arc : cuts_[i].arcs) { coeff += label_arcs.count({arc.from, arc.to}); }
70
71 coeffs[i] = coeff;
72 }
73
74 return coeffs;
