192inline int VRPTW_mcd(int m, int n) {
193 int aux;
194
195 if (m < n) {
196 aux = n;
197 n = m;
198 m = aux;
199 }
200
201 if (n <= 1) return 1;
202
203 while (n != 0) {

205 m = n;
206 n = aux % n;
