39private:
40 Problem *problem;
41 std::priority_queue<BNBNode *, std::vector<BNBNode *>, BNBNodeCompare> bestFirstBNBNodes;

51 void addBNBNode(BNBNode *&node) {
52 if (strategy == BNBNodeSelectionStrategy::BestFirst) {
53 std::lock_guard<std::mutex> lock(bestFirstMutex); // Protect bestFirst queue
54 bestFirstBNBNodes.push(node);
55 } else {
56 otherBNBNodes.enqueue(node);
57 }
58 }
59
60 BNBNode *getNextBNBNode() {
61 BNBNode *node = nullptr;
62
63 if (strategy == BNBNodeSelectionStrategy::BestFirst) {
64 std::lock_guard<std::mutex> lock(bestFirstMutex); // Protect bestFirst queue
65 if (!bestFirstBNBNodes.empty()) {
66 node = bestFirstBNBNodes.top();
67 bestFirstBNBNodes.pop();
68 return node;
69 }
70 }
71
72 if (otherBNBNodes.try_dequeue(node)) { return node; }
73
74 return nullptr; // No node available
75 }

99 void updateGlobalBest(double objectiveValue, BNBNode *currentBNBNode, double boundValue) {
100 // Use atomic compare and swap for updating global best objective
101 double prevGlobalBest = globalBestObjective.load(std::memory_order_acquire);
102 while (objectiveValue > prevGlobalBest &&
103 !globalBestObjective.compare_exchange_weak(prevGlobalBest, objectiveValue, std::memory_order_release)) {
104 // Repeat until update is successful
105 }
106
107 if ((std::floor(boundValue) == globalBestObjective.load(std::memory_order_acquire)) && boundValue > 0) {
108 markSolutionFound(); // If bound equals the best objective, mark solution found
109 }
110 }

112 void processBNBNode(BNBNode *currentBNBNode) {
113 if (isSolutionFound()) return; // Early exit if solution already found
114
115 currentBNBNode->start();
116 currentBNBNode->enforceBranching();
117 problem->evaluate(currentBNBNode);
118 double boundValue = problem->bound(currentBNBNode);
119
120 if (isSolutionFound()) return; // Check again after a potentially long operation
121
122 if (currentBNBNode->getPrune() || boundValue < globalBestObjective.load()) {
123 return; // BNBNode can be pruned
124 }
125
126 double objectiveValue = problem->objective(currentBNBNode);
127 updateGlobalBest(objectiveValue, currentBNBNode, boundValue);
128
129 if (isSolutionFound()) return; // Exit if the solution was found during update
130
131 branch(currentBNBNode);
132 }

134 // Solve using multiple threads
135 void solveParallel(size_t numThreads) {
136 std::vector<std::jthread> threads;
137 for (size_t i = 0; i < numThreads; ++i) {
138 threads.emplace_back([this]() {
139 while (!isSolutionFound()) {
140 BNBNode *currentBNBNode = getNextBNBNode();
141 if (currentBNBNode) {
142 processBNBNode(currentBNBNode);
143 } else {
144 std::this_thread::yield(); // Avoid busy-waiting, yield the thread
145 }
146 }
147 });
148 }

170 }
171
172 if (boundValue < globalBestObjective.load()) continue;
173
174 auto objectiveValue = problem->objective(currentBNBNode);
175 print_info("Objective value: {}\n", objectiveValue);
176
177 if (std::abs(boundValue - objectiveValue) < 1e-2) {
178 fmt::print("\n");
179 fmt::print("\033[34m_SOLUTION FOUND \033[0m: {}\n", objectiveValue);
180
181 globalBestObjective.store(objectiveValue, std::memory_order_release);
182 markSolutionFound();
183 break;
184 }
185
186 if (objectiveValue > globalBestObjective.load()) {
187 globalBestObjective.store(objectiveValue, std::memory_order_release);
188 }
189
190 branch(currentBNBNode);
191 }
192 auto end = std::chrono::high_resolution_clock::now();
193 std::chrono::duration<double> elapsed = end - start;
194 print_info("Elapsed time: {:.2f}\n", elapsed.count());
195 }

197 void branch(BNBNode *&node) {
198 problem->branch(node);
199 auto children = node->getChildren();
200 for (auto &child : children) { addBNBNode(child); }
201 }
