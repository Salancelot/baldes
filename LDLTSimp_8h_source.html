76 void analyzePattern_preordered(const CholMatrixType &ap, bool doLDLT) {
77 const StorageIndex size = StorageIndex(ap.rows());
78 m_matrix.resize(size, size);
79 m_parent.resize(size);
80 m_nonZerosPerCol.resize(size);
81
82 // Using std::vector instead of VLAs for dynamic memory allocation
83 std::vector<StorageIndex> tags(size, 0); // dynamically sized array for tags
84
85 for (StorageIndex k = 0; k < size; ++k) {
86 m_parent[k] = -1; // Parent of k is not yet known
87 tags[k] = k; // Mark node k as visited
88 m_nonZerosPerCol[k] = 0; // Count of nonzeros in column k of L
89
90 for (typename CholMatrixType::InnerIterator it(ap, k); it; ++it) {
91 StorageIndex i = it.index();
92 if (i < k) {
93 // Follow path from i to root of the etree, stop at flagged node
94 for (; tags[i] != k; i = m_parent[i]) {
95 if (m_parent[i] == -1) m_parent[i] = k; // Set parent of i
96 m_nonZerosPerCol[i]++; // L(k,i) is nonzero
97 tags[i] = k; // Mark i as visited
98 }
99 }
100 }
101 }
102
103 // Construct Lp index array from m_nonZerosPerCol column counts
104 StorageIndex *Lp = m_matrix.outerIndexPtr();
105 Lp[0] = 0;
106 for (StorageIndex k = 0; k < size; ++k) { Lp[k + 1] = Lp[k] + m_nonZerosPerCol[k]; }
107
108 // Resize non-zeros for the matrix
109 m_matrix.resizeNonZeros(Lp[size]);
110
111 m_isInitialized = true;
112 m_info = Success;
113 m_analysisIsOk = true;
114 m_factorizationIsOk = false;
115 }

121 template <bool DoLDLT, bool NonHermitian>
122 void factorize_preordered(const CholMatrixType &ap) {
123
124 const StorageIndex size = StorageIndex(ap.rows());
125 const StorageIndex *Lp = m_matrix.outerIndexPtr();
126 StorageIndex *Li = m_matrix.innerIndexPtr();
127 Scalar *Lx = m_matrix.valuePtr();
128
129 // Use aligned arrays for SIMD optimization
130 std::vector<Scalar, Eigen::aligned_allocator<Scalar>> y(size, Scalar(0));
131 std::vector<StorageIndex, Eigen::aligned_allocator<StorageIndex>> pattern(size, 0);
132 std::vector<StorageIndex, Eigen::aligned_allocator<StorageIndex>> tags(size, 0);
133
134 bool ok = true;
135 m_diag.resize(size);
136
137 for (StorageIndex k = 0; k < size; ++k) {
138 y[k] = Scalar(0);
139 StorageIndex top = size;
140 tags[k] = k;
141 m_nonZerosPerCol[k] = 0;
142
143 for (typename CholMatrixType::InnerIterator it(ap, k); it; ++it) {
144 StorageIndex i = it.index();
145 if (i <= k) {
146 y[i] += getSymm(it.value());
147
148 Index len;
149 for (len = 0; tags[i] != k; i = m_parent[i]) {
150 pattern[len++] = i;
151 tags[i] = k;
152 }
153
154 while (len > 0) { pattern[--top] = pattern[--len]; }
155 }
156 }
157
158 // Numerical solve for the current row
159 DiagonalScalar d = getDiag(y[k]) * m_shiftScale + m_shiftOffset;
160 y[k] = Scalar(0); // Reset Y(k)
161
162 // SIMD on dense vector y, but avoid SIMD for sparse indices in Li
163 for (; top < size; ++top) {
164 Index i = pattern[top];
165 Scalar yi = y[i];
166 y[i] = Scalar(0);
167
168 Scalar l_ki = yi / getDiag(m_diag[i]);
169
170 Index p2 = Lp[i] + m_nonZerosPerCol[i];
171
172 // Handle the sparse reductions using regular loops (since Li and Lx are not contiguous)
173 for (Index p = Lp[i]; p < p2; ++p) { y[Li[p]] -= getSymm(Lx[p]) * yi; }
174
175 d -= getDiag(l_ki * getSymm(yi));
176 Li[p2] = k;
177 Lx[p2] = l_ki;
178 ++m_nonZerosPerCol[i];
179 }
180
181 m_diag[k] = d;
182 if (d == RealScalar(0)) {
183 ok = false;
184 break;
185 }
186 }
187
188 m_info = ok ? Success : NumericalIssue;
189 m_factorizationIsOk = true;
190 }

194 permute_symm_to_symm(const MatrixType &mat,
195 SparseMatrix<typename MatrixType::Scalar, DstOrder, typename MatrixType::StorageIndex> &_dest,
196 const typename MatrixType::StorageIndex *perm) {
197 using StorageIndex = typename MatrixType::StorageIndex;
198 using Scalar = typename MatrixType::Scalar;
199 SparseMatrix<Scalar, DstOrder, StorageIndex> &dest(_dest.derived());
200 using VectorI = Matrix<StorageIndex, Dynamic, 1>;
201 using MatEval = internal::evaluator<MatrixType>;
202 using MatIterator = typename internal::evaluator<MatrixType>::InnerIterator;
203
204 enum {
205 SrcOrder = MatrixType::IsRowMajor ? RowMajor : ColMajor,
206 StorageOrderMatch = int(SrcOrder) == int(DstOrder),
207 DstMode = DstOrder == RowMajor ? (DstMode_ == Upper ? Lower : Upper) : DstMode_,
208 SrcMode = SrcOrder == RowMajor ? (SrcMode_ == Upper ? Lower : Upper) : SrcMode_
209 };
210
211 MatEval matEval(mat);
212 Index size = mat.rows();
213 VectorI count(size);
214 count.setZero();
215 dest.resize(size, size);
216
217 const bool isLower = int(SrcMode) == int(Lower);
218 const bool isUpper = int(SrcMode) == int(Upper);
219 const bool isDstLower = int(DstMode) == int(Lower);
220
221 // Precompute permutation
222 std::vector<StorageIndex> perm_cache(size);
223 for (StorageIndex j = 0; j < size; ++j) { perm_cache[j] = perm ? perm[j] : j; }
224
225 // First pass: Count the non-zero elements for each column/row
226 for (StorageIndex j = 0; j < size; ++j) {
227 StorageIndex jp = perm_cache[j];
228
229 for (MatIterator it(matEval, j); it; ++it) {
230 StorageIndex i = it.index();
231 if ((isLower && i < j) || (isUpper && i > j)) continue;
232
233 StorageIndex ip = perm_cache[i];
234
235 // Minimize conditional checks
236 StorageIndex min_ip_jp = std::min(ip, jp);
237 StorageIndex max_ip_jp = std::max(ip, jp);
238
239 count[isDstLower ? min_ip_jp : max_ip_jp]++;
240 }
241 }
242
243 // Allocate space based on the counted non-zero entries
244 dest.outerIndexPtr()[0] = 0;
245 for (Index j = 0; j < size; ++j) { dest.outerIndexPtr()[j + 1] = dest.outerIndexPtr()[j] + count[j]; }
246 dest.resizeNonZeros(dest.outerIndexPtr()[size]);
247
248 // Reset counts for actual filling
249 for (Index j = 0; j < size; ++j) { count[j] = dest.outerIndexPtr()[j]; }
250
251 // Main loop: Populate the destination sparse matrix
252 for (StorageIndex j = 0; j < size; ++j) {
253 StorageIndex jp = perm_cache[j];
254 //__builtin_prefetch(&dest.innerIndexPtr()[count[j]], 1, 1); // Prefetch for insertion
255
256 for (MatIterator it(matEval, j); it; ++it) {
257 StorageIndex i = it.index();
258 if ((isLower && i < j) || (isUpper && i > j)) continue;
259
260 StorageIndex ip = perm_cache[i];
261
262 // Minimize conditional checks
263 StorageIndex min_ip_jp = std::min(ip, jp);
264 StorageIndex max_ip_jp = std::max(ip, jp);
265
266 Index k = count[isDstLower ? min_ip_jp : max_ip_jp]++;
267 dest.innerIndexPtr()[k] = isDstLower ? max_ip_jp : min_ip_jp;
268
269 // Prefetch values for efficient memory access
270 //__builtin_prefetch(&dest.valuePtr()[k], 1, 1);
271
272 if (!StorageOrderMatch) std::swap(ip, jp);
273 if ((isDstLower && ip < jp) || (!isDstLower && ip > jp)) {
274 dest.valuePtr()[k] = NonHermitian ? it.value() : numext::conj(it.value());
275 } else {
276 dest.valuePtr()[k] = it.value();
277 }
278 }
279 }

282 template <int Mode, bool NonHermitian, typename MatrixType, int DestOrder>
283 void permute_symm_to_fullsymm(
284 const MatrixType &mat,
285 SparseMatrix<typename MatrixType::Scalar, DestOrder, typename MatrixType::StorageIndex> &_dest,
286 const typename MatrixType::StorageIndex *perm) {
287
288 using StorageIndex = typename MatrixType::StorageIndex;
289 using Scalar = typename MatrixType::Scalar;
290 using Dest = SparseMatrix<Scalar, DestOrder, StorageIndex>;
291 using VectorI = Matrix<StorageIndex, Dynamic, 1>;
292 using MatEval = internal::evaluator<MatrixType>;
293 using MatIterator = typename internal::evaluator<MatrixType>::InnerIterator;
294
295 MatEval matEval(mat);
296 Dest &dest(_dest.derived());
297
298 enum { StorageOrderMatch = int(Dest::IsRowMajor) == int(MatrixType::IsRowMajor) };
299
300 Index size = mat.rows();
301 VectorI count(size);
302 count.setZero();
303 dest.resize(size, size);
304
305 // First pass: Count non-zeros for each column
306 for (Index j = 0; j < size; ++j) {
307 Index jp = perm ? perm[j] : j;
308
309 for (MatIterator it(matEval, j); it; ++it) {
310 Index i = it.index();
311 Index r = it.row();
312 Index c = it.col();
313 Index ip = perm ? perm[i] : i;
314
315 if (Mode == int(Upper | Lower)) {
316 count[StorageOrderMatch ? jp : ip]++;
317 } else if (r == c) {
318 count[ip]++;
319 } else if ((Mode == Lower && r > c) || (Mode == Upper && r < c)) {
320 count[ip]++;
321 count[jp]++;
322 }
323 }
324 }
325
326 Index nnz = count.sum();
327
328 // Resize for non-zeros and fill outer index
329 dest.resizeNonZeros(nnz);
330 dest.outerIndexPtr()[0] = 0;
331
332 // Unrolling this loop for small loop performance gain
333 for (Index j = 0; j < size; ++j) { dest.outerIndexPtr()[j + 1] = dest.outerIndexPtr()[j] + count[j]; }
334
335 // Reset count for actual insertion
336 for (Index j = 0; j < size; ++j) { count[j] = dest.outerIndexPtr()[j]; }
337
338 // Second pass: Copy data into destination matrix
339 for (StorageIndex j = 0; j < size; ++j) {
340 for (MatIterator it(matEval, j); it; ++it) {
341 StorageIndex i = internal::convert_index<StorageIndex>(it.index());
342 Index r = it.row();
343 Index c = it.col();
344
345 StorageIndex jp = perm ? perm[j] : j;
346 StorageIndex ip = perm ? perm[i] : i;
347
348 if (Mode == int(Upper | Lower)) {
349 Index k = count[StorageOrderMatch ? jp : ip]++;
350 dest.innerIndexPtr()[k] = StorageOrderMatch ? ip : jp;
351 dest.valuePtr()[k] = it.value();
352 } else if (r == c) {
353 Index k = count[ip]++;
354 dest.innerIndexPtr()[k] = ip;
355 dest.valuePtr()[k] = it.value();
356 } else if (((Mode & Lower) == Lower && r > c) || ((Mode & Upper) == Upper && r < c)) {
357 if (!StorageOrderMatch) std::swap(ip, jp);
358 Index k = count[jp]++;
359 dest.innerIndexPtr()[k] = ip;
360 dest.valuePtr()[k] = it.value();
361 k = count[ip]++;
362 dest.innerIndexPtr()[k] = jp;
363 dest.valuePtr()[k] = (NonHermitian ? it.value() : numext::conj(it.value()));
364 }
365 }
366 }

415 Eigen::VectorXd solve(const MatrixBase<Rhs> &b) const {
416 eigen_assert(m_isInitialized && "Decomposition not initialized");
417
418 Eigen::VectorXd dest;
419
420 // Apply forward permutation (if needed)
421 if (m_P.size() > 0) {
422 dest = m_P * b;
423 } else {
424 dest = b;
425 }
426
427 const Scalar regularizationFactor = Scalar(1e-5); // Small regularization term
428
429 // Solve L * y = P * b using SparseSolveTriangular for lower triangular matrix
430 solveTriangular<decltype(matrixL()), decltype(dest), Lower>(matrixL(), dest, regularizationFactor);
431
432 // Solve D * z = y (element-wise division)
433 // Use Eigen's array-based division for vectorized operations
434 for (Index i = 0; i < dest.size(); ++i) { dest(i) /= m_diag(i); }
435 // Solve U * x = z using SparseSolveTriangular for upper triangular matrix
436 solveTriangular<decltype(matrixU()), decltype(dest), Upper>(matrixU(), dest, regularizationFactor);
437
438 // Apply backward permutation (if needed)
439 if (m_Pinv.size() > 0) { dest = m_Pinv * dest; }
440
441 return dest;
